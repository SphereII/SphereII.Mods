31 public static bool CanDamage(EntityAlive self, Entity target)
32 {
33 if (target.IsDead()) return false;
34
35 // Enemy animals can not follow these rules. We also can't use CanDamageEntity here,
36 // because that has a Harmony patch that calls this method again. Instead, reproduce the
37 // logic in EntityEnemyAnimal.CanDamageEntity.
38 if (IsEnemyInAnimalsFaction(self))
39 return target != null && target.entityClass != self.entityClass;
40
41 // Don't damage vehicles if they are immune.
42 if (IsDamageImmuneVehicle(self, target))
43 return false;
44
45 // Don't damage your leader or fellow followers.
46 var myLeader = EntityUtilities.GetLeaderOrOwner(self.entityId);
47 if (IsAllyOfLeader(myLeader, target))
48 return CanDamageAlly(self, target);
49
50 // If two players are involved (directly or as leaders), determine whether they or their
51 // followers can damage each other from the "Player Killing" setting.
52 var selfPlayer = GetPlayerLeader(self, myLeader);
53 var targetPlayer = GetPlayerLeader(target);
54 if (selfPlayer != null && targetPlayer != null)
55 {
56 // FriendlyFireCheck returns true if the players can damage each other
57 return selfPlayer.FriendlyFireCheck(targetPlayer);
58 }
59
60 // If you are a player, don't damage your followers. Otherwise, use factions.
61 if (self is EntityPlayer)
62 {
63 if (IsAlly(target, self))
64 return CanDamageAlly(target, self);
65
66 return !IsFriendlyFireByFaction(self, target);
67 }
68
69 // You can always damage your revenge target, even if it's a player (since they hit first).
70 if (IsCurrentRevengeTarget(self, target))
71 return true;
72
73 // Otherwise, if the target is a player, you can only damage them if you or your
74 // (not necessarily player) leader hate them.
75 if (target is EntityPlayer)
76 {
77 return myLeader == null
78 ? IsEnemyByFaction(self, target)
79 : IsEnemyByFaction(myLeader, target);
80 }
81
82 // You can damage them if they are fighting your leader or allies.
83 if (IsFightingFollowers(myLeader, target))
84 return true;
85
86 // If you have a leader, check friendly fire using their faction.
87 // In all other cases, use the faction relationship between yourself and the target.
88 return myLeader == null
89 ? !IsFriendlyFireByFaction(self, target)
90 : !IsFriendlyFireByFaction(myLeader, target);
91 }

115 public static bool CanTakeDamage(EntityAlive self, Entity damagingEntity)
116 {
117 if (self == null) return true;
118
119 // If it's dead, offer no such protection.
120 if (self.IsDead()) return true;
121
122 // If the damage was not caused by a living entity, take that damage.
123 if (!(damagingEntity is EntityAlive livingEntity))
124 return true;
125
126 return CanDamage(livingEntity, self);

135 public static float GetFriendlyFireRelationship(Entity entity)
136 {
137 // Don't damage neutral NPCs. This is to avoid cases where the player or another NPC steps
138 // in to help neutral human NPCs fighting an enemy - we don't want them to hurt that NPC.
139 if (entity is EntityNPC)
140 return (float)FactionManager.Relationship.Neutral;
141
142 // For all other NPCs (animals, etc.) attackers can damage them if they don't love them.
143 return (float)FactionManager.Relationship.Love;

208 public static bool IsAllyOfParty(Entity self, Entity targetEntity)
209 {
210 if (self == null || targetEntity == null)
211 return false;
212
213 // Don't assume either entity's leader is a player for this check
214 var selfLeader = EntityUtilities.GetLeaderOrOwner(self.entityId) ?? self;
215 var targetLeader = EntityUtilities.GetLeaderOrOwner(targetEntity.entityId) ?? targetEntity;
216
217 if (selfLeader.entityId == targetLeader.entityId)
218 return true;
219
220 // FriendlyFireCheck returns true if friendly fire is allowed
221 if (selfLeader is EntityPlayer selfPlayer && targetLeader is EntityPlayer targetPlayer)
222 return !selfPlayer.FriendlyFireCheck(targetPlayer);
223
224 return false;
225 }

233 public static bool IsEnemy(EntityAlive self, Entity target)
234 {
235 if (self == null) return true;
236
237 if (!(target is EntityAlive targetEntity))
238 return false;
239
240 if (targetEntity.IsDead())
241 return false;
242
243 // Don't start fights with vehicles.
244 if (IsDamageImmuneVehicle(self, target))
245 return false;
246
247 // Don't make enemies out of your followers, your leader, or fellow followers.
248 var myLeader = EntityUtilities.GetLeaderOrOwner(self.entityId);
249 if (IsAllyOfLeader(myLeader ?? self, target))
250 return false;
251
252 // If two players are involved (directly or as leaders), determine whether they or their
253 // followers can damage each other from the "Player Killing" setting.
254 // This is to make sure damage-immune entities are NOT enemies - the reverse is not true.
255 // Just because you can damage them does not make them enemies.
256 var selfPlayer = GetPlayerLeader(self, myLeader);
257 var targetPlayer = GetPlayerLeader(target);
258 // FriendlyFireCheck returns true if the players can damage each other
259 if (selfPlayer != null && targetPlayer != null && !selfPlayer.FriendlyFireCheck(targetPlayer))
260 return false;
261
262 // Our current revenge target is always an enemy.
263 if (IsCurrentRevengeTarget(self, target))
264 return true;
265
266 // If they are fighting my leader or allies, they're an enemy.
267 if (IsFightingFollowers(myLeader, target))
268 return true;
269
270 // If it's an enemy animal, we can't use faction targeting. Instead, test to see if it's
271 // the attack target. (We considered the revenge target, above.)
272 if (IsEnemyInAnimalsFaction(self))
273 {
274 var attackTarget = self.GetAttackTarget();
275 return attackTarget != null && attackTarget.entityId == target.entityId;
276 }
277
278 // They are an enemy if we hate them, or if our leader hates them.
279 return myLeader == null
280 ? IsEnemyByFaction(self, target)
281 : IsEnemyByFaction(myLeader, target);
282 }

290 public static bool IsFightingFollowers(Entity leader, Entity target)
291 {
292 if (leader == null || target == null)
293 return false;
294
295 var theirTarget = GetAggressionTarget(target);
296 if (theirTarget != null)
297 {
298 // Are they fighting the leader?
299 if (theirTarget.entityId == leader.entityId)
300 return true;
301
302 // Are they fighting the leader's followers?
303 var theirTargetLeader = EntityUtilities.GetLeaderOrOwner(theirTarget.entityId);
304 if (theirTargetLeader != null && theirTargetLeader.entityId == leader.entityId)
305 return true;
306 }
307
308 // Is the leader fighting them?
309 var leaderTarget = GetAggressionTarget(leader);
310 return leaderTarget != null && leaderTarget.entityId == target.entityId;
311 }

323 public static bool IsFriend(EntityAlive self, Entity target)
324 {
325
326 if (!(target is EntityAlive targetEntity))
327 return false;
328
329 if (targetEntity.IsDead())
330 return false;
331
332 if (self.entityId == target.entityId)
333 return true;
334
335 // Note: We can't use CanDamage here, because depending upon future features,
336 // it might be possible to damage your friends too.
337
338 var myLeader = EntityUtilities.GetLeaderOrOwner(self.entityId);
339 if (IsAllyOfLeader(myLeader, target) || IsAlly(target, self))
340 return true;
341
342 // If two players are involved (directly or as leaders), determine whether they and their
343 // followers are freinds from the "Player Killing" setting.
344 var selfPlayer = GetPlayerLeader(self, myLeader);
345 var targetPlayer = GetPlayerLeader(target);
346 if (selfPlayer != null && targetPlayer != null)
347 {
348 // FriendlyFireCheck returns true if the players can damage each other
349 return !selfPlayer.FriendlyFireCheck(targetPlayer);
350 }
351
352 // Don't consider revenge targets to be our friends.
353 if (IsCurrentRevengeTarget(self, target))
354 return false;
355
356 // Targets who are attacking our allies are not friends.
357 if (IsFightingFollowers(myLeader, target))
358 return false;
359
360 // They are a friend if we love them, or our leader loves them.
361 return myLeader == null
362 ? IsFriendlyFireByFaction(self, target)
363 : IsFriendlyFireByFaction(myLeader, target);
364 }

372 public static bool IsEnemyByFaction(Entity self, Entity target)
373 {
374 if (!(self is EntityAlive livingSelf && target is EntityAlive livingTarget))

404 public static bool IsFriendlyFireByFaction(Entity self, Entity target)
405 {
406 if (!(self is EntityAlive livingSelf && target is EntityAlive livingTarget))
407 return false;
408
409 // If we have the "damage relationship" cvar, use that value. Otherwise, use the default.
410 float damageRelationship = GetFriendlyFireRelationship(target);
411
412 if (livingSelf.Buffs.HasCustomVar(DamageRelationshipCVarName))
413 {
414 var buffValue = livingSelf.Buffs.GetCustomVar(DamageRelationshipCVarName);
415 // GetCustomVar also sets the custom var to zero if it doesn't exist; check that.
416 if (buffValue > 0)
417 damageRelationship = buffValue;
418 }
419
420 var relationship = EntityUtilities.GetFactionRelationship(livingSelf, livingTarget);
421 return relationship >= damageRelationship;
422 }

530 private static Entity GetAggressionTarget(Entity attacker)
531 {
