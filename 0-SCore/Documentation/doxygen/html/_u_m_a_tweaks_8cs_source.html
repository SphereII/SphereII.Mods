16 // Disable the pregeneration of UMAs, since this can be time consuming and resource intensive, causing menu lag, etc.
17 // Create the directory if it doesn't exist, so when they are populated on demand, they'll be saved.
18 [HarmonyPatch(typeof(Archetypes))]
19 [HarmonyPatch("PregenStatic")]
20 public class SphereII_PregenStatic
21 {
22 public static bool Prefix(Archetype __instance, ref List<Archetype> ___archetypes)
23 {
24 // Check if this feature is enabled.
25 if (!Configuration.CheckFeatureStatus(AdvFeatureClass, Feature))
26 return true;
27
28 // Create the folder if it doesn't exist
29 string strUMAFolder = Path.Combine(Application.dataPath, "..", "Data", "UMATextures");
30 if (!Directory.Exists(strUMAFolder))
31 Directory.CreateDirectory(strUMAFolder);
32
33 // No not pre-generating the UMAs. Let them be created on the fly when needed, but saved for future use.
34 return false;
35 }
36 }
