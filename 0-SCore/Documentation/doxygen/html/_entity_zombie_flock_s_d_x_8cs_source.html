566 {
567 var result = Vector3.zero;
568 if (blockList == null) return result;
569 if (blockList.Length == 0) return result;
570 for (var i = (int)position.x - checkRange; i <= position.x + checkRange; i++)
571 for (var j = (int)position.z - checkRange; j <= position.z + checkRange; j++)
572 for (var k = (int)position.y - checkRange; k <= position.y + checkRange; k++)
573 {
574 var blockF = GameManager.Instance.World.GetBlock(i, k, j);
575 if (Array.Exists(blockList, s => s.Equals(Block.list[blockF.type].GetBlockName())))
576 {
577 result = new Vector3(i, k, j);
578 return result;
579 }
580 }

622 // flocking logic
623 private void flockTasks()
624 {
625 if (GamePrefs.GetBool(EnumGamePrefs.DebugStopEnemiesMoving) || GameStats.GetInt(EnumGameStats.GameState) == 2) return;
626 if (world.IsRemote()) return;
627 if (IsDead()) return;
628 if (masterEntity == null && !hasFlock && maxToSpawn > 0)
629 SpawnFlock();
630 else if (masterEntity == null && hasFlock && oldmasterID > 0) FindOldMaster();
631 GetEntitySenses().ClearIfExpired();
632 if (AttackTimeout > 0) AttackTimeout--;
633 if (AttackTimeout <= 0)
634 {
635 var a = Waypoint - position;
636 var sqrMagnitude = a.sqrMagnitude;
637 if (sqrMagnitude < 1f || sqrMagnitude > 6400f)
638 {
639 if (!isWithinHomeDistanceCurrentPosition() && GetMasterEntity() == null && !isHunting)
640 {
641 // uses vanilla code to stay near "home position" if its a "master"
642 var ye = RandomPositionGenerator.CalcTowards(this, 2 * getMaximumHomeDistance(), 2 * getMaximumHomeDistance(), 2 * getMaximumHomeDistance(),
643 getHomePosition().position.ToVector3());
644 if (!ye.Equals(Vector3.zero))
645 {
646 if (debug) Debug.Log("Going Home");
647 Waypoint = ye;
648 HasWaypoint = true;
649 }
650 }
651 else
652 {
653 HasWaypoint = false;
654 if (base.GetRevengeTarget() != null && (base.GetRevengeTarget().GetDistanceSq(this) < 6400f && UnityEngine.Random.value <= 0.5f || isHunting))
655 {
656 // if it's targeting an enemy. Notice that if it's hunting I just want it to get it done.
657 Waypoint = base.GetRevengeTarget().GetPosition() + Vector3.up;
658 }
659 else
660 {
661 if (GetMasterEntity() == null)
662 {
663 // if it finds a target block nearby, it will go for it. Not attack it, just go near it
664 // this will make them "destroy" crops for example, if we make them target crop blocks.
665 if (FindLandSpot())
666 {
667 // going for landing spot
668 Waypoint = landPosition + new Vector3((float)((rand.RandomDouble * 2.0 - 1.0) * 3.0), (float)((rand.RandomDouble * 2.0 - 1.0) * 3.0),
669 (float)((rand.RandomDouble * 2.0 - 1.0) * 3.0));
670 }
671 else
672 {
673 // chooses a random waypoint - vanilla code
674 Waypoint = GetPosition() + new Vector3((float)((rand.RandomDouble * 2.0 - 1.0) * 16.0), (float)((rand.RandomDouble * 2.0 - 1.0) * 16.0),
