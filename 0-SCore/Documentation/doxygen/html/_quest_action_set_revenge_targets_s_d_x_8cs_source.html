92 public override void PerformAction(Quest ownerQuest)
93 {
94 var targetEntities = GetTargetEntities(ownerQuest);
95 if (targetEntities == null)
96 return;
97
98 var entitiesInRange = GetEntitiesInRange(ownerQuest);
99 if (entitiesInRange == null)
100 return;
101
102 foreach (var entity in entitiesInRange)
103 {
104 if (entity is EntityPlayer)
105 continue;
106
107 // Don't set the revenge targets of entities hired by the quest owner,
108 // or by entities hired by members of the quest owner's party
109 if (EntityTargetingUtilities.IsAllyOfParty(entity, ownerQuest.OwnerJournal.OwnerPlayer))
110 continue;
111
112 var target = GetRandomTarget(targetEntities);
113 entity.SetRevengeTarget(target);
114 entity.SetRevengeTimer(REVENGE_TICKS);
115 }
116 }

118 private Bounds GetBounds(Quest ownerQuest)
119 {
120 float distance = DEFAULT_DISTANCE;
121 Vector3 center = ownerQuest.OwnerJournal.OwnerPlayer.position;
122
123 if ("location".EqualsCaseInsensitive(Value))
124 {
125 var size = ownerQuest.GetLocationSize();
126 if (size != Vector3.zero)
127 {
128 // The center should be the center of the POI, not the player position
129 var location = ownerQuest.GetLocation();
130
131 center = new Vector3(
132 location.x + size.x / 2f,
133 location.y + size.y / 2f,
134 location.z + size.z / 2f);
135
136 // The bounds constructor shrinks the size vector in half, compensate here
137 return new Bounds(center, 2f * size);
138 }
139 }
140 else if (!string.IsNullOrEmpty(Value) && !float.TryParse(Value, out distance))
141 {
142 return default;
143 }
144
145 return new Bounds(center, new Vector3(distance, distance, distance));
146 }

166 private EntityAlive[] GetTargetEntities(Quest ownerQuest)
167 {
168 var targetEntities = new List<EntityAlive>() { ownerQuest.OwnerJournal.OwnerPlayer };
169
170 if (ID == null || ID.EqualsCaseInsensitive(OWNER_ID))
171 return targetEntities.ToArray();
172
173 if (ID.EqualsCaseInsensitive(SHARED_ID))
174 {
175 var shared = ownerQuest.GetSharedWithIDList();
176 if (shared == null)
177 return targetEntities.ToArray();
178
179 var notInRange = ownerQuest.GetSharedWithListNotInRange();
180
181 foreach (var entityid in shared)
182 {
183 var entity = GameManager.Instance.World.GetEntity(entityid) as EntityPlayer;
184 if ( entity == null ) continue;
185 if (notInRange != null && notInRange.Contains(entity))
186 continue;
187
188 targetEntities.Add(entity);
189 }
190 return targetEntities.ToArray();
191 }
192
193 // "owner" and "party" are the only allowed ID values
194 return null;
195 }
