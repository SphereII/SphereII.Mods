112 {
113
114 switch (_commandName)

124 GameManager.ShowTooltip(_player as EntityPlayerLocal, Localization.Get("debugcontrol_turnoff"));
125 break;
126 case "debugcontrol_turnonWater":
127 Manager.BroadcastPlay(_blockPos.ToVector3() + Vector3.one * 0.5f, this.activateSound, 0f);
128 UpdateValves(_blockPos, true);
129 GameManager.ShowTooltip(_player as EntityPlayerLocal, Localization.Get("debugcontrol_turnonWater"));
130 break;
131 case "debugcontrol_turnoffWater":
132 Manager.BroadcastPlay(_blockPos.ToVector3() + Vector3.one * 0.5f, this.activateSound, 0f);
133 UpdateValves(_blockPos, false);

138 Manager.BroadcastPlay(_blockPos.ToVector3() + Vector3.one * 0.5f, this.activateSound, 0f);
139 GameManager.ShowTooltip(_player as EntityPlayerLocal, Localization.Get("debugcontrol_startingbot"));

142 }
