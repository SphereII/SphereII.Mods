138 // This method is a modified version of vanilla, doing the same checks and balances. However, we do use the player position a bit more, and we change which biome spawning group we
139 // will use, when below the terrain.
140 public static void SpawnUpdate(string _spawnerName, bool _bSpawnEnemyEntities, ChunkAreaBiomeSpawnData _chunkBiomeSpawnData, ref List<Entity> spawnNearList, ref int lastClassId)
141 {
142 if (_chunkBiomeSpawnData == null) return;
143 if (_bSpawnEnemyEntities)
144 {
145 if (GameStats.GetInt(EnumGameStats.EnemyCount) >= GamePrefs.GetInt(EnumGamePrefs.MaxSpawnedZombies))
146 _bSpawnEnemyEntities = false;
147 else if (GameManager.Instance.World.aiDirector.BloodMoonComponent.BloodMoonActive) _bSpawnEnemyEntities = false;
148 }
149
150 if (!_bSpawnEnemyEntities && GameStats.GetInt(EnumGameStats.AnimalCount) >= GamePrefs.GetInt(EnumGamePrefs.MaxSpawnedAnimals)) return;
151 var flag = false;
152 var players = GameManager.Instance.World.GetPlayers();
153
154 // Player Position.
155 var position = Vector3.zero;
156 var rect = new Rect(1, 1, 1, 1);
157 foreach (var t in players)
158 {
159 if (!t.Spawned) continue;
160
161 position = t.GetPosition();
162 rect = new Rect(position.x - 40f, position.z - 40f, 80f, 20f);
163 if (!rect.Overlaps(_chunkBiomeSpawnData.area)) continue;
164
165 flag = true;
166 break;
167 }
168
169 // No valid player position.
170 if (position == Vector3.zero)
171 return;
172
173 // Don't allow above ground spawning.
174 var playerPosition = new Vector3i(position);
175 float offSet = GameManager.Instance.World.GetTerrainHeight(playerPosition.x, playerPosition.z);
176 if (offSet <= playerPosition.y)
177 return;
178
179 var minDistance = _bSpawnEnemyEntities ? 28 : 48;
180 var maxDistance = _bSpawnEnemyEntities ? 54 : 70;
181 //if (!flag || !FindRandomSpawnPointNearPositionUnderground(rect, minDistance, maxDistance, false, out var vector, playerPosition))
182 var size = new Vector3(60, 10, 60);
183 if (!flag || !GameManager.Instance.World.FindRandomSpawnPointNearPositionUnderground(playerPosition, 16, out int x, out int y, out int z, size ))
184 return;
185
186 var vector = new Vector3();
187 vector.x = x;
188 vector.z = z;
189 vector.y = y;
190 // Mob is above terrain; ignore.
191 if (vector.y > offSet)
192 return;
193
194
195 var biome = GameManager.Instance.World.Biomes.GetBiome(_chunkBiomeSpawnData.biomeId);
196 if (biome == null) return;
197
198 // Customize which spawning.xml entry to we want to use for spawns.
199 var caveType = "Cave";
200 // Search for the biome_Cave spawn group. If not found, load the generic Cave one.
201 var biomeSpawnEntityGroupList = BiomeSpawningClass.list[biome.m_sBiomeName + "_Cave"];
202 if (biomeSpawnEntityGroupList == null) biomeSpawnEntityGroupList = BiomeSpawningClass.list["Cave"];
203 // if we are below 30, look for the biome specific deep cave, then deep cave if its not set.
204 if (vector.y < 30)
205 {
206 caveType = "DeepCave";
207 biomeSpawnEntityGroupList = BiomeSpawningClass.list[biome.m_sBiomeName + "_DeepCave"];
208 if (biomeSpawnEntityGroupList == null) biomeSpawnEntityGroupList = BiomeSpawningClass.list["DeepCave"];
209 }
210
211 if (biomeSpawnEntityGroupList == null)
212 return;
213
214 var edaytime = GameManager.Instance.World.IsDaytime() ? EDaytime.Day : EDaytime.Night;
215 var gameRandom = GameManager.Instance.World.GetGameRandom();
216 string entityGroupName = null;
217 var num = -1;
218 var num2 = gameRandom.RandomRange(biomeSpawnEntityGroupList.list.Count);
219 var j = 0;
220
221 while (j < 5)
222 {
223 BiomeSpawnEntityGroupData biomeSpawnEntityGroupData2 = biomeSpawnEntityGroupList.list[num2];
224 if (biomeSpawnEntityGroupData2.daytime == EDaytime.Any || biomeSpawnEntityGroupData2.daytime == edaytime)
225 {
226 bool flag2 = EntityGroups.IsEnemyGroup(biomeSpawnEntityGroupData2.entityGroupRefName);
227 if (!flag2 || _bSpawnEnemyEntities)
228 {
229 int num4 = biomeSpawnEntityGroupData2.maxCount;
230 if (flag2)
231 {
232 num4 = EntitySpawner.ModifySpawnCountByGameDifficulty(num4);
233 }
234
235 entityGroupName = biomeSpawnEntityGroupData2.entityGroupRefName + "_" + biomeSpawnEntityGroupData2.daytime.ToStringCached<EDaytime>();
236 ulong respawnDelayWorldTime = _chunkBiomeSpawnData.GetRespawnDelayWorldTime(entityGroupName);
237 if (respawnDelayWorldTime > 0UL)
238 {
239 if (GameManager.Instance.World.worldTime < respawnDelayWorldTime)
240 {
241 break;
242 }
243 _chunkBiomeSpawnData.ClearRespawn(entityGroupName);
244 }
245 if (_chunkBiomeSpawnData.GetEntitiesSpawned(entityGroupName) < num4)
246 {
247
248 num = num2;
249 break;
250 }
251 }
252 }
253 j++;
254 num2 = (num2 + 1) % biomeSpawnEntityGroupList.list.Count;
255 }
256
257 if (num < 0)
258 return;
259 var bb = new Bounds(vector, new Vector3(4f, 2.5f, 4f));
260 GameManager.Instance.World.GetEntitiesInBounds(typeof(Entity), bb, spawnNearList);
261 var count = spawnNearList.Count;
262 spawnNearList.Clear();
263 if (count > 0)
264 return;
265 var biomeSpawnEntityGroupData3 = biomeSpawnEntityGroupList.list[num];
266 var randomFromGroup = EntityGroups.GetRandomFromGroup(biomeSpawnEntityGroupData3.entityGroupRefName, ref lastClassId);
267 var spawnDeadChance = biomeSpawnEntityGroupData3.spawnDeadChance;
268 _chunkBiomeSpawnData.IncEntitiesSpawned(entityGroupName);
269 var entity = EntityFactory.CreateEntity(randomFromGroup, vector);
270 entity.SetSpawnerSource(EnumSpawnerSource.Dynamic, _chunkBiomeSpawnData.chunk.Key, entityGroupName);
271 var myEntity = entity as global::EntityAlive;
272 if (myEntity) myEntity.SetSleeper();
273
274 // Debug.Log("Spawning: " + myEntity.entityId + " " + vector );
275 GameManager.Instance.World.SpawnEntityInWorld(entity);
276
277
278 if (spawnDeadChance > 0f && gameRandom.RandomFloat < spawnDeadChance) entity.Kill(DamageResponse.New(true));
279 GameManager.Instance.World.DebugAddSpawnedEntity(entity);
280 }
