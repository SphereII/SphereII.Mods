108 var PathingCode = PathingCubeParser.GetValue(_signText, "pc");
109 var setLeader = PathingCubeParser.GetValue(_signText, "leader");
110
111 if (Task.ToLower() == "stay")
112 entity.Buffs.AddBuff("buffOrderStay");
113 if (Task.ToLower() == "wander")
114 entity.Buffs.AddBuff("buffOrderWander");
115 if (Task.ToLower() == "guard")
116 // Use the buff that issues the "guard" order, not the one that issues the "stay" order
117 entity.Buffs.AddBuff("buffOrderGuard");
118 if (Task.ToLower() == "follow")
119 entity.Buffs.AddBuff("buffOrderFollow");
120
121 foreach (var buff in Buff.Split(','))
122 entity.Buffs.AddBuff(buff);
123
124 // Set up the pathing cube if available
125 if (!string.IsNullOrEmpty(PathingCode))
126 {
127 if (StringParsers.TryParseFloat(PathingCode, out var pathingCode))
128 entity.Buffs.SetCustomVar("PathingCode", pathingCode);
129 }
130
131 // We are using the tile entity to transfer the owner ID from the client to the player.
132 var tileEntity = GameManager.Instance.World.GetTileEntity(0, _blockPos) as TileEntityPoweredTrigger;
133 if (tileEntity != null)
134 {
135 var persistentPlayerList = GameManager.Instance.GetPersistentPlayerList();
136 var playerData = persistentPlayerList.GetPlayerData(tileEntity.GetOwner());
137 if (playerData != null)
138 OwnerID = playerData.EntityId;
139
140 // Set up ownership, but only after the entity is spawned.
141 if (OwnerID > 0 && !string.IsNullOrEmpty(setLeader))
142 {
143 EntityUtilities.SetLeaderAndOwner(entity.entityId, (int)OwnerID, false);
144 }
