21internal class BlockTriggeredSDX : BlockLoot
22{
23 private static readonly string AdvFeatureClass = "AdvancedTileEntities";
24 private int RandomIndex;
25
26 public override void Init()
27 {
28 base.Init();
29 if (Properties.Values.ContainsKey("RandomIndex"))
30 RandomIndex = StringParsers.ParseSInt32(Properties.Values["RandomIndex"], 0, -1, NumberStyles.Any);
31 }
32
33 public override string GetActivationText(WorldBase _world, BlockValue _blockValue, int _clrIdx, Vector3i _blockPos, EntityAlive _entityFocusing)
34 {
35 if (!_blockValue.Block.Properties.Values.ContainsKey("ActivateOnLook")) return "";
36 var activateOnLook = StringParsers.ParseBool(_blockValue.Block.Properties.Values["ActivateOnLook"]);
37 if (!activateOnLook) return "";
38 AdvLogging.DisplayLog(AdvFeatureClass, _blockValue.Block.GetBlockName() + ": Activating Block on GetActivationText");
39
40 ActivateBlock(_world, _clrIdx, _blockPos, _blockValue, true, true);
41
42 return "";
43 }
44
45 // don't open the loot container.
46 public override bool OnBlockActivated(WorldBase _world, int _cIdx, Vector3i _blockPos, BlockValue _blockValue, EntityAlive _player)
47 {
48 if (!_blockValue.Block.Properties.Values.ContainsKey("IsContainer")) return true;
49 var isContainer = StringParsers.ParseBool(_blockValue.Block.Properties.Values["IsContainer"]);
50 if (!isContainer) return true;
51 base.OnBlockActivated(_world, _cIdx, _blockPos, _blockValue, _player);
52 return true;
53
54 }
55
56 public override bool ActivateBlock(WorldBase _world, int _cIdx, Vector3i _blockPos, BlockValue _blockValue, bool isOn, bool isPowered)
57 {
58 // If there's no transform, no sense on keeping going for this class.
59 var ebcd = _world.GetChunkFromWorldPos(_blockPos).GetBlockEntity(_blockPos);
60 if (ebcd == null || ebcd.transform == null)
61 return false;
62
63 var componentsInChildren = ebcd.transform.GetComponentsInChildren<Animator>();
64 if (componentsInChildren == null) return true;
65 for (var i = componentsInChildren.Length - 1; i >= 0; i--)
66 {
67 var animator = componentsInChildren[i];
68
69 AdvLogging.DisplayLog(AdvFeatureClass, _blockValue.Block.GetBlockName() + ": Animator: " + animator.name + " : Active: " + isOn);
70 if (isOn)
71 {
72 var random = Random.Range(0, RandomIndex);
73 AdvLogging.DisplayLog(AdvFeatureClass, _blockValue.Block.GetBlockName() + ": Random Index for " + animator.name + " Value: " + random);
74
75 animator.SetInteger("RandomIndex", random);
76 AdvLogging.DisplayLog(AdvFeatureClass, _blockValue.Block.GetBlockName() + ": Setting Bool for On: True " + animator.name);
77 animator.SetBool("On", true);
78 AdvLogging.DisplayLog(AdvFeatureClass, _blockValue.Block.GetBlockName() + ": Trigger for On: " + animator.name);
79 animator.SetTrigger("TriggerOn");
80 }
81 else
82 {
83 AdvLogging.DisplayLog(AdvFeatureClass, _blockValue.Block.GetBlockName() + ": Setting Bool for On: false" + animator.name);
84 animator.SetBool("On", false);
85 // AdvLogging.DisplayLog(AdvFeatureClass, _blockValue.Block.GetBlockName() + ": Turning Off Animator " + animator.name);
86 // animator.enabled = false;
87 }
88 }
89
90 return true;
91 }
