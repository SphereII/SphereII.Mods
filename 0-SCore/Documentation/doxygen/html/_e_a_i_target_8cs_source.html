1using System;
2using HarmonyLib;

4namespace Harmony.EAI

6 [HarmonyPatch(typeof(EAITarget))]
7 [HarmonyPatch("check")]
8 public class EAITarget_check
9 {
10 public static void Postfix(ref bool __result, EAITarget __instance, global::EntityAlive _e)
11 {
12 if (!__result) return;
13
14 // Only "humans" and entities with Use Factions tags should use faction-based targeting
15 if (!EntityUtilities.IsHuman(__instance.theEntity.entityId)
16 && !EntityUtilities.UseFactions(__instance.theEntity))
17 return;
18
19 // If its a vehicle.. umm.. no.
20 if (_e is EntityVehicle)
21 {
22 __result = false;
23 return;
24 }
25
26 // Checks if we are allies, either share a leader, or is our leader.
27 if (EntityTargetingUtilities.IsAlly(__instance.theEntity, _e))
28 {
29 __result = false;
30 return;
31 }
32
33
34 // Do we have a revenge target? Are they the ones attacking us?
35 if (EntityTargetingUtilities.IsCurrentRevengeTarget(__instance.theEntity, _e))
36 {
37 __result = true;
38 return;
39 }
40
41 // If the target entity is attacking our ally, target them too.
42 var leader = EntityUtilities.GetLeaderOrOwner(__instance.theEntity.entityId);
43 if (EntityTargetingUtilities.IsFightingFollowers(leader, _e))
44 {
45 __result = true;
46 return;
47 }
48
49 // allow an override for the action stuff. If its set here, then use faction targeting.
50 if (!EntityUtilities.CheckProperty(__instance.theEntity.entityId, "AllEntitiesUseFactionTargeting"))
51 {
52 // This patches the check for all entities, so if we're not supposed to use faction
53 // targeting for everything, stop now. Otherwise, do the faction check.
54 if (!Configuration.CheckFeatureStatus("AdvancedNPCFeatures", "AllEntitiesUseFactionTargeting"))
55 {
56 // Even if *all* entities don't use faction targeting, this entity still should
57 // if it has one of the UseFactions tags. Only stop now if it doesn't.
58 if (!EntityUtilities.UseFactions(__instance.theEntity))
59 {
60 return;
61 }
62 }
63 }
64 var myRelationship = FactionManager.Instance.GetRelationshipTier(__instance.theEntity, _e);
65 switch (myRelationship)
66 {
67 case FactionManager.Relationship.Hate:
68 __result = true;
69 break;
70 case FactionManager.Relationship.Dislike:
71 __result = true;
72 break;
73 case FactionManager.Relationship.Neutral:
74 case FactionManager.Relationship.Like:
75 case FactionManager.Relationship.Love:
76 case FactionManager.Relationship.Leader:
77 default:
78 __result = false;
79 break;
80 }
81
82 return;
83 }
84 }
