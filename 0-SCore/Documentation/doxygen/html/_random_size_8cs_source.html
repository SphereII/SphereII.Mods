62
63 if (__instance is EntityPlayerLocal)
64 return;
65
66
67 if (RandomSizeHelper.AllowedRandomSize(__instance))

69 // This is the distributed random heigh multiplier. Add or adjust values as you see fit. By default, it's just a small adjustment.
70 var numbers = new float[9] { 0.7f, 0.8f, 0.9f, 0.9f, 1.0f, 1.0f, 1.0f, 1.1f, 1.2f };
71 var random = new Random();
72
73 var randomIndex = random.Next(0, numbers.Length);
74 var flScale = numbers[randomIndex];
75
76 AdvLogging.DisplayLog(AdvFeatureClass, " Random Size: " + flScale);
77 __instance.Buffs.AddCustomVar("RandomSize", flScale);
78 // scale down the zombies, or upscale them
79 __instance.gameObject.transform.localScale = new Vector3(flScale, flScale, flScale);
80 }
81
82 // Check if there's random ranges
83 var entityClass = __instance.EntityClass;
84 if (entityClass.Properties.Values.ContainsKey("RandomSizes"))
85 {
86 var Ranges = new List<float>();
87 var flScale = 1f;
88 foreach (var text in entityClass.Properties.Values["RandomSizes"].Split(','))
89 Ranges.Add(StringParsers.ParseFloat(text));
90
91 var random = new Random();
92 var randomIndex = random.Next(0, Ranges.Count);
93 flScale = Ranges[randomIndex];
94 AdvLogging.DisplayLog(AdvFeatureClass, " Random Size: " + flScale);
