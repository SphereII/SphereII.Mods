20 public override bool GetBindingValue(ref string value, string bindingName)
21 {
22 if (bindingName == null) return base.GetBindingValue(ref value, bindingName);
23
24
25 // These bindings only exist under our special class
26 if (this.entityAliveSDX)
27 {
28 var faction = EntityUtilities.GetFactionRelationship(entityAliveSDX, player);
29
30 // We'll use this later.
31 var leader = EntityUtilities.GetLeaderOrOwner(entityAliveSDX.entityClass) as EntityAlive;
32 switch (bindingName)
33 {
34 case "isdislike":
35 value = "false";
36 if (faction <= (int)FactionManager.Relationship.Dislike)
37 value = "true";
38 break;
39 case "isneutral":
40 value = "false";
41 if (faction >= (int)FactionManager.Relationship.Neutral)
42 value = "true";
43 break;
44 case "islike":
45 value = "false";
46 if (faction >= (int)FactionManager.Relationship.Like)
47 value = "true";
48 break;
49
50 case "npcnametitle":
51 value = Localization.Get(entityAliveSDX.Title);
52 break;
53 case "npcfirstname":
54 value = Localization.Get(entityAliveSDX.FirstName);
55 break;
56 case "npcfaction":
57 value = Localization.Get(FactionManager.Instance.GetFaction(entityAliveSDX.factionId).Name);
58 break;
59 case "npclevel":
60 value = entityAliveSDX.Progression.GetLevel().ToString();
61 break;
62 case "npchealth":
63 value = entityAliveSDX.Health.ToString();
64 break;
65 case "npcmaxhealth":
66 value = entityAliveSDX.GetMaxHealth().ToString();
67 break;
68 case "npcexptonextlevel": // I'm not sure if this stuff is hooked up yet
69 value = entityAliveSDX.Progression.ExpToNextLevel.ToString();
70 break;
71 case "npcexpfornextlevel": // same, not sure if its here.
72 value = entityAliveSDX.Progression.GetExpForNextLevel().ToString();
73 break;
74 case "npcleader": // if leader is null, just break, otherwise send the name back
75 if (leader == null) break;
76 value = leader.EntityName; // return sphereii
77 break;
78 case "npcleaderid": // returns 171
79 if (leader == null) break;
80 value = leader.entityId.ToString();
81 break;
82 case "npccurrentholdingitem":
83 value = entityAliveSDX.inventory.holdingItem.GetLocalizedItemName();
84 break;
85 case "npcarmor": // not sure if this is hooked up.
86 value = EffectManager.GetValue(PassiveEffects.PhysicalDamageResist, null, 0f, entityAliveSDX, null, default(FastTags), true, true, true, true, 1, true).ToString();
87 break;
88 case "npccurrentorder":
89 value = EntityUtilities.GetCurrentOrder(entityAliveSDX.entityId).ToString();
90 break;
91 case "npcxp":
92 if (entityAliveSDX?.Progression.ExpDeficit > 0)
93 {
94 float b = Math.Max(this.lastDeficitValue, 0f) * 1.01f;
95 value = this.bindingXp.Format(b);
96 this.currentValue = (float)this.entityAliveSDX.Progression?.ExpDeficit / (float)this.entityAliveSDX.Progression?.GetExpForNextLevel();
97 if (this.currentValue != this.lastDeficitValue)
98 {
99 this.lastDeficitValue = Mathf.Lerp(this.lastDeficitValue, this.currentValue, Time.deltaTime * this.xpFillSpeed);
100 if (Mathf.Abs(this.currentValue - this.lastDeficitValue) < 0.005f)
101 {
102 this.lastDeficitValue = this.currentValue;
103 }
104 }
105 }
106 else
107 {
108 float v2 = Math.Max(this.lastValue, 0f) * 1.01f;
109 value = this.bindingXp.Format(v2);
110 this.currentValue = entityAliveSDX.Progression.GetLevelProgressPercentage();
111 if (this.currentValue != this.lastValue)
112 {
113 this.lastValue = Mathf.Lerp(this.lastValue, this.currentValue, Time.deltaTime * this.xpFillSpeed);
114 if (Mathf.Abs(this.currentValue - this.lastValue) < 0.005f)
115 {
116 this.lastValue = this.currentValue;
117 }
118 }
119 }
120
121 break;
122 case "npcxpcolor":
123 if (entityAliveSDX.Progression?.ExpDeficit > 0)
124 {
125 value = this.expDeficitColor;
126 }
127 else
128 {
129 value = ((this.currentValue == this.lastValue) ? this.standardXPColor : this.updatingXPColor);
130 }
131 break;
132 case "npcbagusedslots":
133 value = this.entityAliveSDX.bag.GetUsedSlotCount().ToString();
134
135 break;
136 case "npccarrycapacity":
137 // Grab the Carry Capacity.
138 var unlocked = this.playerCarryCapacityFormatter.Format((int)EffectManager.GetValue(PassiveEffects.CarryCapacity, null, 0f, entityAliveSDX, null, default(FastTags), true, true, true, true, 1, true));
139
140 // Grab the Bag size
141 var totalslot = this.playerBagSizeFormatter.Format((int)EffectManager.GetValue(PassiveEffects.BagSize, null, 0f, entityAliveSDX, null, default(FastTags), true, true, true, true, 1, true));
142
143 // By default, we'll set then unlocked as the right value to show
144 value = unlocked;
145
146 // If nothing is in total slot, that means its not initialized yet, so just return, using the unlocked value.
147 if (string.IsNullOrEmpty(totalslot))
148 break;
149
150 // Convert the strings to a number and see which is bigger.
151 if (StringParsers.ParseSInt32(unlocked) > StringParsers.ParseSInt32(totalslot))
152 value = totalslot;
153 else
154 value = unlocked;
155 break;
156 case "npcbagfreeslots":
157 var total = StringParsers.ParseSInt32(this.playerBagSizeFormatter.Format((int)EffectManager.GetValue(PassiveEffects.BagSize, null, 0f, this.entityAliveSDX, null, default(FastTags), true, true, true, true, 1, true)));
158 var used = StringParsers.ParseSInt32(this.entityAliveSDX.bag.GetUsedSlotCount().ToString());
159 var freeslots = total - used;
160 value = freeslots.ToString();
161 break;
162 case "npcbagfill":
163 value = Mathf.Clamp01((float)this.entityAliveSDX.bag.GetUsedSlotCount() / (float)this.entityAliveSDX.bag.SlotCount).ToCultureInvariantString();
164 break;
165 case "npcbagfillcolor":
166 var usedslots = this.entityAliveSDX.bag.GetUsedSlotCount().ToString();
167 var totalslot2 = this.playerCarryCapacityFormatter.Format((int)EffectManager.GetValue(PassiveEffects.CarryCapacity, null, 0f, this.entityAliveSDX, null, default(FastTags), true, true, true, true, 1, true));
168
169 value = usedslots;
170
171 if (string.IsNullOrEmpty(totalslot2))
172 break;
173
174 if (StringParsers.ParseSInt32(usedslots) > StringParsers.ParseSInt32(totalslot2))
175 value = usedslots;
176 else
177 value = totalslot2;
178
179 var used2 = StringParsers.ParseFloat(usedslots);
180 var total2 = StringParsers.ParseFloat(totalslot2);
181
182 var percent = used2 / total2;
183
184 value = "43,124,18,255"; // Green
185 if (percent > 0.75)
186 value = "255,255,0,255"; // Yellow
187 if (percent > 0.90)
188 value = "255,144,24,255"; // Orange
189 if (percent > 1)
190 value = "175,30,25,255"; // Red
191 break;
192
193 // From sphereii - Custom Code - "Free's" NPC Portrait for use.
194 case "npcportrait":
195 if (base.xui.Dialog.Respondent != null)
196 {
197 value = this.NPC.NPCInfo.Portrait;
198 }
199 if (string.IsNullOrWhiteSpace(value))
200 value = "npcPortraitGeneric";
201 break;
202 // From XUiC_SkillListWindow
203 case "skillpointsavailable":
204 string v = this.pointsAvailable;
205 value = this.skillPointsAvailableFormatter.Format(v, entityAliveSDX.Progression.SkillPoints);
206 break;
207 case "hasskillpoint":
208 if (entityAliveSDX.Progression.SkillPoints > 0)
209 value = "true";
210 else
211 value = "false";
212
213 break;
214 default:
215 value = "";
216 return true;
217
218 }
219
220 }
221
222 return true;
223 }

225 public override void Update(float _dt)
226 {
227 if (base.ViewComponent.IsVisible && Time.time > this.updateTime)
228 {
229 this.updateTime = Time.time + 0.25f;
230 if ( entityAliveSDX)
231 base.RefreshBindings(this.IsDirty);
232 if (this.IsDirty)
233 {
234 this.IsDirty = false;
235 }
236 }
237 base.Update(_dt);
238 }
