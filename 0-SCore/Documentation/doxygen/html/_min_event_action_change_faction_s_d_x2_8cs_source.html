13 public override void Execute(MinEventParams _params)
14 {
15 for (var i = 0; i < targets.Count; i++)
16 {
17 var entity = targets[i];
18 if (entity != null)
19 {
20 // If the faction name is original, try to find the original faction of the entity, stored via cvar.
21 if (Faction == "original")
22 {
23 // If there's already a factionoriginal cvar, retrive the faction name to be re-assigned.
24 if (entity.Buffs.HasCustomVar("FactionOriginal"))
25 {
26 var FactionID = (byte)entity.Buffs.GetCustomVar("FactionOriginal");
27 var Temp = FactionManager.Instance.GetFaction(FactionID);
28 if (Temp != null)
29 {
30 Debug.Log("Changed " + entity.EntityName + " faction from " + entity.factionId + " to " + Temp.ID);
31
32 Faction = Temp.Name;
33 }
34 }
35 else
36 {
37 if (FactionManager.Instance.GetFactionByName(entity.EntityName).ID == 0)
38 {
39 Debug.Log("Created new faction for " + entity.EntityName + ": " + entity.factionId);
40
41 entity.factionId = FactionManager.Instance.CreateFaction(entity.EntityName).ID;
42 entity.factionRank = byte.MaxValue;
43 }
44 }
45 }
46
47 // Search for the faction
48 var newFaction = FactionManager.Instance.GetFactionByName(Faction);
49 if (newFaction != null)
50 {
51 // If there's no original cvar, store it here. That's because it doesn't seem to persist over restarts, but cvars will
52 if (!entity.Buffs.HasCustomVar("FactionOriginal"))
53 entity.Buffs.SetCustomVar("FactionOriginal", entity.factionId);
54
55 // Set the new faction ID as a cvar for saving.
56 entity.Buffs.SetCustomVar("FactionNew", newFaction.ID);
57
58 Debug.Log("Changing " + entity.EntityName + " faction from " + entity.factionId + " to " + newFaction.ID);
59
60 entity.factionId = newFaction.ID;
61 Debug.Log("\nNew Faction: " + entity.factionId);
62 }
63 entity.SetAttackTarget(null, 0);
64 entity.SetRevengeTarget(null);
65 }
66 }
67 }
