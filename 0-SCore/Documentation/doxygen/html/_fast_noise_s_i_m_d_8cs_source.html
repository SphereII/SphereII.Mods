572 public VectorSet(Vector3[] vectors, int sampleSizeX = -1, int sampleSizeY = -1, int sampleSizeZ = -1, int samplingScale = 0)
573 {
574 var vectorSetArray = new float[vectors.Length * 3];
575
576 for (var i = 0; i < vectors.Length; i++)
577 {
578 vectorSetArray[i] = vectors[i].x;
579 vectorSetArray[i + vectors.Length] = vectors[i].y;
580 vectorSetArray[i + vectors.Length * 2] = vectors[i].z;
581 }
582
583 nativePointer = NewVectorSet(vectorSetArray, vectorSetArray.Length, samplingScale, sampleSizeX, sampleSizeY, sampleSizeZ);
584 }
585
