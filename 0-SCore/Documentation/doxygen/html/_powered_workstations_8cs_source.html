24
25 // Read the block properties whenever we check, since we don't really persists whether or not we need power.
26 var block = GameManager.Instance.World.GetBlock(myTileEntity.ToWorldPos());
27 var block2 = Block.list[block.type];
28 blNeedsPower = false;
29 if (block2.Properties.Values.ContainsKey("RequirePower"))
30 blNeedsPower = StringParsers.ParseBool(block2.Properties.Values["RequirePower"]);
31
32 AdvLogging.DisplayLog(AdvFeatureClass, "Require Power? : " + blNeedsPower);
33
34 // Doesn't need any power, so don't check.
35 if (blNeedsPower == false)
36 return true;
37
38 AdvLogging.DisplayLog(AdvFeatureClass, "Workstation requires power. Checking near by tile entities.");
39
40 // Check the nearby chunks for Power source tile entities, and check if they are on.
41 var blockPosition = myTileEntity.ToWorldPos();
42 var world = GameManager.Instance.World;
43 var num = World.toChunkXZ(blockPosition.x);
44 var num2 = World.toChunkXZ(blockPosition.z);
45 for (var i = -1; i < 2; i++)
46 for (var j = -1; j < 2; j++)
47 {
48 var chunk = (Chunk)world.GetChunkSync(num + j, num2 + i);
49 if (chunk == null) continue;
50 var tileEntities = chunk.GetTileEntities();
51 foreach (var t in tileEntities.list)
52 {
53 AdvLogging.DisplayLog(AdvFeatureClass, "\tDetected Tile Entity: " + t);
54 var tileEntity = t as TileEntityPowerSource;
55 if (tileEntity == null) continue;
56 if (tileEntity.IsActive(world))
57 AdvLogging.DisplayLog(AdvFeatureClass, "\tEntity is Active " + t);
58 if (tileEntity.IsOn)
59 {
60 AdvLogging.DisplayLog(AdvFeatureClass, "\tTile Entity is On: " + t);
61 var distanceSq = (blockPosition.ToVector3() - tileEntity.ToWorldPos().ToVector3()).sqrMagnitude;
62 if (!(distanceSq <= 20 * 20)) continue;
63 AdvLogging.DisplayLog(AdvFeatureClass, myTileEntity + " Found Power Source as Fuel with " + tileEntity);
64 //myTileEntity.HasPowerSDX = true;
65 return true;
66 }
67
68 AdvLogging.DisplayLog(AdvFeatureClass, "\tTile Entity is off: " + t);
69 }
70 }
71
72 return false;
