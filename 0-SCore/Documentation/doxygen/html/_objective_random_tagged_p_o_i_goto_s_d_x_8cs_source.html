59 public override void ParseProperties(DynamicProperties properties)
60 {
61 if (properties.Values.ContainsKey(PropIncludeTags))
62 {
63 IncludeTags = POITags.Parse(properties.Values[PropIncludeTags].Replace(" ", ""));
64 }
65 if (properties.Values.ContainsKey(PropExcludeTags))
66 {
67 ExcludeTags = POITags.Parse(properties.Values[PropExcludeTags].Replace(" ", ""));
68 }
69 if (properties.Values.ContainsKey(PropDistance))
70 {
71 string searchDistance = properties.Values[PropDistance].Replace(" ", "");
72 if (searchDistance.Contains("-"))
73 {
74 string[] distances = searchDistance.Split(new char[] { '-' });
75 MinSearchDistance = StringParsers.ParseFloat(distances[0]);
76 MaxSearchDistance = StringParsers.ParseFloat(distances[1]);
77 }
78 else
79 {
80 MaxSearchDistance = StringParsers.ParseFloat(searchDistance);
81 }
82 // Remove the property value so the base class doesn't parse it
83 properties.Values.Remove(PropDistance);
84 }
85 base.ParseProperties(properties);

233 public static Vector3 PrefabCenterToPosition(Vector2 prefabCenter)
234 {
235 // Since the center is a 2D vector, its "y" is actually the z-axis position
236 int x = (int)prefabCenter.x;
237 int y = (int)GameManager.Instance.World.GetHeightAt(prefabCenter.x, prefabCenter.y);
238 int z = (int)prefabCenter.y;
239 return new Vector3(x, y, z);
