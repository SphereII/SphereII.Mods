77 public static void Postfix(TileEntity __instance, ref bool __result, World world)
78 {
79 var block = GameManager.Instance.World.GetBlock(__instance.ToWorldPos());
80 var block2 = Block.list[block.type];
81 var isAlwaysActive = false;
82
83 if (!block2.Properties.Values.ContainsKey("AlwaysActive")) return;
84
85 isAlwaysActive = StringParsers.ParseBool(block2.Properties.Values["AlwaysActive"]);
86
87 if (!isAlwaysActive) return;
88
89 __result = true;
90 AdvLogging.DisplayLog(AdvFeatureClass, block2.GetBlockName() + ": TileEntity is Active.");
91 var blCanTrigger = false;
92
93 var bounds = 6;
94 if (block2.Properties.Values.ContainsKey("ActivationDistance"))
95 bounds = StringParsers.ParseSInt32(block2.Properties.Values["ActivationDistance"]);
96
97 // Scan for the player in the radius as defined by the Activation distance of the block
98 var entitiesInBounds = GameManager.Instance.World.GetEntitiesInBounds(null,
99 new Bounds(__instance.ToWorldPos().ToVector3(), Vector3.one * 20));
100 if (entitiesInBounds.Count > 0)
101 {
102 AdvLogging.DisplayLog(AdvFeatureClass,
103 block2.GetBlockName() + ": TileEntity has Entities in Bound of " + bounds);
104 foreach (var t in entitiesInBounds)
105 {
106 var player = t as EntityPlayer;
107 if (!player) continue;
108 var distance = (player.position - __instance.ToWorldPos().ToVector3()).sqrMagnitude;
109 if (distance > block2.GetActivationDistanceSq())
110 continue;
111
112 AdvLogging.DisplayLog(AdvFeatureClass,
113 block2.GetBlockName() + ": Player: " + player.EntityName + " is in bounds");
114
115 if (block2.Properties.Values.ContainsKey("ActivationBuffs"))
116 {
117 foreach (var strbuff in block2.Properties.Values["ActivationBuffs"].Split(','))
118 {
119 AdvLogging.DisplayLog(AdvFeatureClass,
120 block2.GetBlockName() + ": Checking ActivationBuffs: " + strbuff);
121 var strBuffName = strbuff;
122 var checkValue = -1f;
123
124 // Check if there's a ( ) at the end of the buff; this will be used as a cvar hook.
125 var start = strbuff.IndexOf('(');
126 var end = strbuff.IndexOf(')');
127 if (start != -1 && end != -1 && end > start + 1)
128 {
129 checkValue = StringParsers.ParseFloat(strbuff.Substring(start + 1, end - start - 1));
130 strBuffName = strbuff.Substring(0, start);
131 AdvLogging.DisplayLog(AdvFeatureClass,
132 block2.GetBlockName() + ": Actviation Buff is a cvar: " + strBuffName +
133 ". Requires value: " + checkValue);
134 }
135
136 // If the player has a buff by this name, trigger it.
137 if (player.Buffs.HasBuff(strBuffName))
138 {
139 AdvLogging.DisplayLog(AdvFeatureClass,
140 block2.GetBlockName() + ": Buff has been found: " + strBuffName);
141 blCanTrigger = true;
142 }
143
144 // If the player has a cvar, then do some extra checks
145 if (!player.Buffs.HasCustomVar(strBuffName)) continue;
146
147
148 AdvLogging.DisplayLog(AdvFeatureClass,
149 block2.GetBlockName() + ": Cvar has been found: " + strBuffName);
150 // If there's no cvar value specified, just allow it.
151 if (checkValue == -1)
152 {
153 AdvLogging.DisplayLog(AdvFeatureClass,
154 block2.GetBlockName() + ": Cvar found, and does not require a specific value.");
155 blCanTrigger = true;
156 }
157
158 // If a cvar is set, then check to see if it matches
159 if (checkValue > -1)
160 {
161 AdvLogging.DisplayLog(AdvFeatureClass,
162 block2.GetBlockName() + ": Cvar found, and requires a value of " + checkValue +
163 " Player has: " + player.Buffs.GetCustomVar(strBuffName));
164 if (player.Buffs.GetCustomVar(strBuffName) == checkValue)
165 blCanTrigger = true;
166 }
167
168 // If any of these conditions match, trigger the Activate block
169 if (blCanTrigger)
170 {
171 AdvLogging.DisplayLog(AdvFeatureClass,
172 block2.GetBlockName() + ": Checks successfully passed for player: " +
173 player.EntityName);
174 break;
175 }
176
177 AdvLogging.DisplayLog(AdvFeatureClass,
178 block2.GetBlockName() + ": Checks were NOT success for player: " + player.EntityName);
179 }
180 }
181 else
182 blCanTrigger = true;
183
184 if (blCanTrigger)
185 break;
186 }
187 }
188
189 if (blCanTrigger)
190 {
191 AdvLogging.DisplayLog(AdvFeatureClass,
192 block2.GetBlockName() + ": TileEntity can call ActivateBlock. Calling it...");
193 Block.list[block.type].ActivateBlock(world, __instance.GetClrIdx(), __instance.ToWorldPos(), block,
194 true, true);
195 }
196 else
197 {
198 AdvLogging.DisplayLog(AdvFeatureClass,
199 block2.GetBlockName() + ": TileEntity is Active but is not Activating.");
200 Block.list[block.type].ActivateBlock(world, __instance.GetClrIdx(), __instance.ToWorldPos(), block,
201 false, true);
202 }
203
204 __result = true;
205 return;
206 }
