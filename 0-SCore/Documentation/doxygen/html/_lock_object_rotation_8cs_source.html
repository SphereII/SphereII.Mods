29 private void MouseMovement()
30 {
31 // If we are holding down the left mouse button...
32 if (Input.GetMouseButton(0) || mouseMovementOn)
33 {
34 // Grab the new position of the mouse, 0,0 - 1,1
35 var newPosition = uiCam.ScreenToViewportPoint(Input.mousePosition);
36 //Vector3 newPosition = Camera.main.ScreenToViewportPoint(Input.mousePosition);
37
38 // Compute the delta in the x and y positions. Do inversion for the yDelta. Note X/Y are flipped.
39 var horDelta = -(newPosition.x - mousePosition.x) * speed * Time.deltaTime;
40 var verDelta = (newPosition.y - mousePosition.y) * speed * Time.deltaTime;
41
42 // Make sure we don't go over our max range
43 if (GetAngle(transform.localEulerAngles.x) + verDelta > startX + maxX || GetAngle(transform.localEulerAngles.x) + verDelta < startX - maxX)
44 verDelta = 0;
45 if (GetAngle(transform.localEulerAngles.y) + horDelta > startY + maxY || GetAngle(transform.localEulerAngles.y) + horDelta < startY - maxY)
46 horDelta = 0;
47
48 // Compute & assign the new angle
49 var newAngles = new Vector3(transform.localEulerAngles.x + verDelta,
50 transform.localEulerAngles.y + horDelta, transform.localEulerAngles.z);
51 transform.localEulerAngles = newAngles;
52 }
