98 public static void Postfix(UAIPackage _package, XElement _element)
99 {
100 if (_element == null || _package == null)
101 return;
102
103 if (!_element.HasAttribute("name"))
104 return;
105
106 var name = _element.GetAttribute("name");
107
108 var actions = _package.GetActions();
109 if (actions == null)
110 return;
111
112 for (int i = 0; i < actions.Count; i++)
113 {
114 var action = actions[i];
115 if (action.Name == name)
116 {
117 SCoreUtils.StoreAttributes(_package, action, _element);
118 break;
119 }
120 }
