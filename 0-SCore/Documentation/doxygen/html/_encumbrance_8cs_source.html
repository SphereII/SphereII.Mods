20 public static float CalculateSlots(ItemStack[] slots)
21 {
22 // By default, set everything as 0.1 weight (pounds? ounces? magic? )
23 var minimumEncumberance = float.Parse(Configuration.GetPropertyValue(AdvFeatureClass, "MinimumWeight"));
24
25 var flTotalEncumbrance = 0f;
26 int num = 0;
27 while (num < slots.Length && num < slots.Length)
28 {
29 if (slots[num].IsEmpty())
30 {
31 num++;
32 continue;
33 }
34
35
36 float itemWeight = minimumEncumberance;
37
38 // Check if the item value has an item weight attached to it, and if so, use that number instead.
39 if (ItemClass.list[slots[num].itemValue.type].Properties.Values.ContainsKey("ItemWeight"))
40 {
41
42 float.TryParse(ItemClass.list[slots[num].itemValue.type].Properties.Values["ItemWeight"], out itemWeight);
43 }
44 else
45 {
46 // if it doesn't have an item weight, check its block entry.
47 var blockValue = slots[num].itemValue.ToBlockValue();
48 if ( blockValue.Block.Properties.Values.ContainsKey("ItemWeight"))
49 float.TryParse(blockValue.Block.Properties.Values["ItemWeight"], out itemWeight);
50 }
51
52 // Calculate the total weight of the stack
53 float flTotalWeight = itemWeight * slots[num].count;
54
55 flTotalEncumbrance += flTotalWeight;
56 num++;
57 }
58 return flTotalEncumbrance;
59 }
60 public static void CheckEncumbrance()
61 {
62 int num = 0;
63
64 var localPlayer = GameManager.Instance.World.GetPrimaryPlayer();
65 if (localPlayer == null) return;
66
67 var flDefaultMax = float.Parse(Configuration.GetPropertyValue(AdvFeatureClass, "MaxEncumbrance"));
68 var flMaxEncumbrance = flDefaultMax;
69 if (localPlayer.Buffs.HasCustomVar("MaxEncumbrance"))
70 flMaxEncumbrance = localPlayer.Buffs.GetCustomVar("MaxEncumbrance");
71
72 // Safety check. If the encumbrance goes to a negative value, reset it.
73 if (flMaxEncumbrance < 0)
74 {
75 Log.Out($"Max Encumbrance is below 0. Using {flDefaultMax}");
76 flMaxEncumbrance = flDefaultMax;
77 }
78 try
79 {
80 ItemStack[] slots = localPlayer.bag.GetSlots();
81 var flTotalEncumbrance = CalculateSlots(slots);
82 AdvLogging.DisplayLog(AdvFeatureClass, $"\tBackpack Total encumbrance: {flTotalEncumbrance} from {slots.Length} Slots.");
83
84 if (Configuration.CheckFeatureStatus(AdvFeatureClass, "Encumbrance_ToolBelt"))
85 {
86 var toolbeltWeight = 0f;
87 ItemStack[] toolbeltSlots = localPlayer.inventory.GetSlots();
88 toolbeltWeight = CalculateSlots(toolbeltSlots);
89 flTotalEncumbrance += toolbeltWeight;
90 AdvLogging.DisplayLog(AdvFeatureClass, $"\tToolbelt Total encumbrance: {toolbeltWeight} from {toolbeltSlots.Length} Slots.");
91 }
92
93 if (Configuration.CheckFeatureStatus(AdvFeatureClass, "Encumbrance_Equipment"))
94 {
95 var equipmentWeight = 0f;
96 var items = localPlayer.equipment.GetItems();
97 foreach( var item in items )
98 {
99 if ( item == null ) continue;
100 if (ItemClass.GetForId(item.type) == null) continue;
101
102 var itemWeight = float.Parse(Configuration.GetPropertyValue(AdvFeatureClass, "MinimumWeight"));
103 if ( item.ItemClass.Properties.Values.ContainsKey("ItemWeight"))
104 float.TryParse(item.ItemClass.Properties.Values["ItemWeight"], out itemWeight);
105 equipmentWeight += itemWeight;
106 AdvLogging.DisplayLog(AdvFeatureClass, $"\tEquipment encumbrance: {item.ItemClass.GetItemName()} Weight: {itemWeight} ");
107 }
108
109 flTotalEncumbrance += equipmentWeight;
110 AdvLogging.DisplayLog(AdvFeatureClass, $"\tEquipment Total encumbrance: {equipmentWeight} from {items.Length} Slots.");
111 }
112
113 // 1 being at Max loaded, and anything above is extra encumberance.
114 float over = flTotalEncumbrance / flMaxEncumbrance;
115
116 var cvar = Configuration.GetPropertyValue(AdvFeatureClass, "EncumbranceCVar");
117 localPlayer.Buffs.AddCustomVar(cvar, over);
118 AdvLogging.DisplayLog(AdvFeatureClass, $"Total encumbrance: {flTotalEncumbrance}");
119
120
121 }
122 catch (Exception ex)
123 {
124 Log.Out($"Encumbrance Mod Error: {ex.ToString()}");
125 }
126 }
127 private static void Postfix(EntityPlayerLocal __instance)

137 __instance.inventory.OnToolbeltItemsChangedInternal += CheckEncumbrance;
138
139 if (Configuration.CheckFeatureStatus(AdvFeatureClass, "Encumbrance_Equipment"))

142 }
