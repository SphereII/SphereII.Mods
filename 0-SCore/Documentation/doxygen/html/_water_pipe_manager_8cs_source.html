184 return false;

191 return true;

197 // Counts how many water blocks are in the surrounding area.
198 public int CountWaterBlocks(Vector3i position, int range = 4)
199 {
200 int count = 0;
201 for (int x = -range; x <= range; x++)
202 {
203 for (int z = -range; z <= range; z++)
204 {
205 for (int y = position.y - range; y <= position.y + range; y++)
206 {
207 var waterCheck = new Vector3i(position.x + x, y, position.z + z);
208 if (GameManager.Instance.World.IsWater(waterCheck))
209 count++;
210 }
211 }
212 }
213 return count;
214 }
