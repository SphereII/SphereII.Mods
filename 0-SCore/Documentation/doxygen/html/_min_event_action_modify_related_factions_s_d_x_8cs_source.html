186 private void ModifyRelatedFactionRelationships(byte targetEntityFactionId, Faction otherFaction)
187 {
188 if (relatedFactions == null) return;
189
190 for (var i = 0; i < relatedFactions.Length; i++)
191 {
192 var relatedFaction = FactionManager.Instance.GetFactionByName(relatedFactions[i]);
193
194 // Guard against users entering a related faction name that doesn't exist, is the
195 // name of a target entity's faction, or is the same as the "faction" attribute
196 if (relatedFaction != null &&
197 relatedFaction.ID != otherFaction.ID &&
198 relatedFaction.ID != targetEntityFactionId)
199 {
200 var modifier = CalculateRelatedFactionModifier(
201 relatedFaction.GetRelationship(otherFaction.ID));
202
203 relatedFaction.ModifyRelationship(targetEntityFactionId, modifier);
204
205 if (debug) Debug.Log(string.Format("relationship to {0} modified by {1}", relatedFactions[i], modifier));
206 }
207 }

210 private float CalculateRelatedFactionModifier(float relationshipToFaction)
211 {
212 float ratio = 1; // default to Leader
213
214 if (relationshipToFaction < 200f) // Hate
215 ratio = -0.6666667f;
216 else if (relationshipToFaction < 400f) // Dislike
217 ratio = -0.3333333f;
218 else if (relationshipToFaction < 600f) // Neutral
219 ratio = 0f;
220 else if (relationshipToFaction < 800f) // Like
221 ratio = 0.3333333f;
222 else if (relationshipToFaction < 1001f) // Love
223 ratio = 0.6666667f;
224
225 var relationshipToTarget = ratio * value * scale;
226
227 // The value 255f has a special meaning in TFP code, so avoid it
228 return relationshipToTarget == 255f ? relationshipToTarget + 1 : relationshipToTarget;
229 }
