127 [HarmonyPatch("Reset")]
128 public class TileEntityLootContainerReset
129 {
130 private static bool IsSupposedToBeLocked(Vector3i position)
131 {
132 // Detect which prefab we are at.
133 var prefabInstance = GameManager.Instance.GetDynamicPrefabDecorator()?.GetPrefabAtPosition(position.ToVector3());
134 if (prefabInstance == null)
135 return false;
136
137 for (var i = 0; i < prefabInstance.prefab.size.x; i++)
138 {
139 for (var j = 0; j < prefabInstance.prefab.size.z; j++)
140 {
141 for (var k = 0; k < prefabInstance.prefab.size.y; k++)
142 {
143 // Find the world position of this block then check if it matches our current position.
144 var num = i + prefabInstance.boundingBoxPosition.x;
145 var num2 = j + prefabInstance.boundingBoxPosition.z;
146 var num7 = World.toBlockY(k + prefabInstance.boundingBoxPosition.y);
147 var localPosition = new Vector3i(num, num7, num2);
148 if (localPosition != position)
149 continue;
150
151 // Grab the blockValue from the original prefab reference, then use its meta value.
152 var blockValue = prefabInstance.prefab.GetBlock(i, k, j);
153 if (blockValue.ischild)
154 continue;
155
156 return (blockValue.meta & 4) > 0;
157 }
158 }
159 }
160
161 return false;
162 }
