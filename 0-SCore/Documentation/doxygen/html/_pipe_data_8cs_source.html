57 public bool FindAllPipes(Pipe _pipe, bool findFirst = true)
58 {
59 // If we exceeded our max, stop searching. This is the limit of the pipe system.
60 if (currentPipe >= maxPipes) return true;
61 currentPipe++;
62
63 // Generate a new set of PipeData. This will hold data on connecting pipes and water sources,
64 // and collect data from all the neighbors.
65 var pipeData = new PipeData(_pipe.GetBlockPos());
66 var neighbors = pipeData.GetNeighbors();
67
69 if (neighbors.IsWaterSource() && findFirst)
70 {
71 // Add the watert source as part of the piping system
72 var waterSource = neighbors.GetWaterSource();
73 var pipe = new Pipe(waterSource);
74 pipe.AddWater(waterSource);
75
76 if (!pipes.ContainsKey(waterSource))
77 pipes.Add(waterSource, pipe);
78 return true;
79 }
80
81 // Loop around the pipes and check if they have already been indexed. If not, keep
82 // searching.
83 foreach (var position in neighbors.GetPipes())
84 {
85 if (pipes.ContainsKey(position)) continue;
86
87 // Register the pipe to the list so we can keep track of it.
88 var pipe = new Pipe(position);
89 pipes.Add(position, pipe);
90
91 // Keep searching if we haven't discovered this pipe yet.
92 var result = FindAllPipes(pipe, findFirst);
93 if (result)
94 return true;
95
96 }
97 return false;
98 }
