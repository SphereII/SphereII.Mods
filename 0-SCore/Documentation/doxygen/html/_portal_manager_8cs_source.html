160 public Vector3i GetDestination(Vector3i source)
161 {
162
163 if (PortalMap.ContainsKey(source))
164 {
165 var sourceName = PortalMap[source];
166 var item = new PortalItem(source, sourceName);
167 if (item.Destination == "NA") return Vector3i.zero;
168
169 // Loop around every teleport position, matchng up the name.
170 foreach (var portal in PortalMap)
171 {
172 var portalItem = new PortalItem(portal.Key, portal.Value);
173 if (item.Destination == portalItem.Source)
174 {
175 // don't teleport to the same location.
176 if (source == portal.Key) continue;
177 return portal.Key;
178 }
179 }
180 return CheckForPrefabLocation(item);
181 }
182
183
184 return Vector3i.zero;

187 public Vector3i CheckForPrefabLocation(PortalItem item)
188 {
189 var prefabInstance = item.GetPrefabInstance();
190 if (prefabInstance == null) return Vector3i.zero;
191
192 // return prefabInstance.boundingBoxPosition;
193 var size = prefabInstance.prefab.size;
194 var first = prefabInstance.boundingBoxPosition;
195 for (int i = 0; i < size.z; i++)
196 {
197 for (int j = 0; j < size.x; j++)
198 {
199 for (int k = 0; k < size.y; k++)
200 {
201 BlockValue blockValue = prefabInstance.prefab.GetBlock(j, k, i);
202 if (blockValue.isair) continue;
203 if (blockValue.Block.HasTileEntity)
204 {
205 if (blockValue.Block.Properties.Values.ContainsKey("Location"))
206 {
207 var location = blockValue.Block.Properties.Values["Location"];
208 int num2 = i + prefabInstance.boundingBoxPosition.z;
209 int num4 = j + prefabInstance.boundingBoxPosition.x;
210 int num5 = k + prefabInstance.boundingBoxPosition.y;
211 var vector3i = new Vector3i(num4, num5, num2);
212 AddPosition(vector3i, location);
213
214 if (first == prefabInstance.boundingBoxPosition)
215 first = vector3i;
216 }
217 }
218 }
219 }
220 }
221 return first;
222 }

225 // Dialog / Buff teleportation.
226 public Vector3i GetDestination(string location)
227 {
228 var destinationItem = new PortalItem(Vector3i.zero, location);
229
230 // If the source and destination is the same, then this means its a legacy teleport location, likely coming from a teleport from minevents where its only one way
231 var legacy = destinationItem.Source == destinationItem.Destination;
232
233 var destination = Vector3i.zero;
234 foreach (var portal in PortalMap)
235 {
236 var portalItem = new PortalItem(portal.Key, portal.Value);
237 if (location == portalItem.Destination)
238 {
239 destination = portal.Key;
240 break;
241 }
242 if (destinationItem.Destination == portalItem.Destination)
243 {
244 destination = portal.Key;
245 break;
246 }
247
248 if (legacy && location == portalItem.Source)
249 {
250 destination = portal.Key;
251 break;
252 }
253 }
254 // If the destination is zero here, check if its a prefab
255 if (destination == Vector3i.zero)
256 {
257 return CheckForPrefabLocation(destinationItem);
258 }
259 // If the portal needs power, make sure its on.
260 var tileEntity = GameManager.Instance.World.GetTileEntity(0, destination) as TileEntityPoweredPortal;
261 if (tileEntity == null) return Vector3i.zero; // Dead portal
262 if (tileEntity.RequiredPower <= 0) return destination;
263 if (tileEntity.IsPowered) return destination;
264 return Vector3i.zero;

295 {
296 PooledExpandableMemoryStream pooledExpandableMemoryStream = (PooledExpandableMemoryStream)_threadInfo.parameter;
