55 private void Current_BlockDestroy(Block block, Vector3i blockPos)
56 {
57 if (base.Complete)
58 {
59 return;
60 }
61
62 var blockname = block.GetBlockName();
63
64 // Allow the objective to be searched via comma-delimited number
65 var matchingName = "";
66
67 bool flag = false;
68 if (base.ID == null || base.ID == "")
69 {
70 flag = true;
71 }
72 else
73 {
74 // If the ID has commas in it, loop around and treat it like a list.
75 foreach( var temp in base.ID.Split(','))
76 {
77 if ( temp.EqualsCaseInsensitive(blockname))
78 {
79 matchingName = temp;
80 flag = true;
81 break;
82 }
83 }
84 if (base.ID.EqualsCaseInsensitive(blockname))
85 {
86 flag = true;
87 }
88 if (blockname.Contains(":") && base.ID.EqualsCaseInsensitive(blockname.Substring(0, blockname.IndexOf(':'))))
89 {
90 flag = true;
91 }
92 }
93 if (!flag && base.ID != null && base.ID != "")
94 {
95 Block blockByName;
96 if ( string.IsNullOrEmpty(matchingName))
97 blockByName = Block.GetBlockByName(base.ID, true);
98 else
99 blockByName = Block.GetBlockByName(matchingName, true);
100
101 if (blockByName != null && blockByName.SelectAlternates && blockByName.ContainsAlternateBlock(blockname))
102 {
103 flag = true;
104 }
105 }
106
107 // If we've reached here, and there's still not a valid block, see if its a tag
108 if ( !flag )
109 {
110 foreach (var tag in base.ID.Split(','))
111 {
112 if (block.HasAnyFastTags(FastTags.Parse(tag)))
113 {
114 flag = true;
115 break;
116 }
117 }
118 }
119
120 if (flag && base.OwnerQuest.CheckRequirements())
121 {
122 AddDestroyedBlock();
123 HandleParty();
124 }
125 }
