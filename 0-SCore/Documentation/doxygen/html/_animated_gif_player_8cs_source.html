332 private IEnumerator Load()
333 {
334 if (FileName.Length == 0)
335 // Debug.LogWarning("File name not set");
336 yield break;
337
338 // Set status to loading
339 State = GifPlayerState.Loading;
340 string path;
341
342
343 // Create file path
344 if (FileName.Substring(0, 4) == "http")
345 {
346 // Get image from web
347 path = FileName;
348 }
349 else
350 {
351 // Local storage
352 string gifPath;
353 switch (Path)
354 {
355 case GifPath.StreamingAssetsPath:
356 gifPath = Application.streamingAssetsPath;
357 break;
358 case GifPath.PersistentDataPath:
359 gifPath = Application.persistentDataPath;
360 break;
361 case GifPath.TemporaryCachePath:
362 gifPath = Application.temporaryCachePath;
363 break;
364 default:
365 gifPath = Application.streamingAssetsPath;
366 break;
367 }
368
369#if !UNITY_2017_2_OR_NEWER
370 // Url encode for Unity 2017 1 or newer
371 path = Uri.EscapeUriString(FileName);
372
373 // Encode #
374 path = path.Replace("#", "%23");
375#endif
376
377#if (UNITY_ANDROID && !UNITY_EDITOR) || (UNITY_WEBGL && !UNITY_EDITOR)
378 path = System.IO.Path.Combine(gifPath, FileName);
379#else
380 path = System.IO.Path.Combine("file://" + gifPath, FileName);
381#endif
382 }
383
384
385#pragma warning disable 618
386 using (var www = new WWW(path.Replace("#", "%23")))
387#pragma warning restore 618
388 {
389 yield return www;
390
391 if (string.IsNullOrEmpty(www.error) == false)
392 // Gif file could not be loaded from streaming assets
393 //Debug.LogWarning("File load error.\n" + www.error + "\nPath:" + WWW.EscapeURL(path, Encoding.Default));
394 State = GifPlayerState.Error;
395 else
396 // Gif file loaded. Pass stream to gif decoder
397 lock (_locker)
398 {
399 if (_gifDecoder.Read(new MemoryStream(www.bytes)) == GifDecoder.Status.StatusOk)
400 {
401 // Gif header was read. Prepare the gif
402 // Set status to preprocessing
403 State = GifPlayerState.PreProcessing;
404
405 // Create the target texture
406 CreateTargetTexture();
407
408 // Show the first frame
409 StartDecoder();
410 }
411 else
412 {
413 // Maybe image?
414 StartCoroutine(DownloadImage(path));
415 // Error decoding gif
416 //Debug.LogWarning("Error loading gif");
417 //State = GifPlayerState.Error;
418 //if (OnLoadError != null) OnLoadError();
419 }
420 }
421 }
422 }

577 // Reads and caches all frames
578 private void BufferFrames()
579 {
580 if (_useThreadedDecoder)
581 {
582 // Threaded
583 _wh.Set(); // Signal thread to read the next frame
584 return;
585 }
586
587 // Not threaded
588 lock (_locker)
589 {
590 while (true)
591 {
592 // Read a single frame
593 _gifDecoder.ReadNextFrame(false);
594
595 if (_gifDecoder.AllFramesRead) break;
596
597 // Get the current frame
598 var frame = _gifDecoder.GetCurrentFrame();
599
600 // Add frame to frame cache
601 AddFrameToCache(frame);
602 }
603
604 _frameIsReady = true;
605 }
606 }
607
608 // Add a frame to the frame cache
609 private void AddFrameToCache(GifDecoder.GifFrame frame)
610 {
611 // Create a coopy of the data to add to the cache since the frame data array is reused for the next frame
612 var copyOfImage = new byte[frame.Image.Length];
613 Buffer.BlockCopy(frame.Image, 0, copyOfImage, 0, frame.Image.Length);
614 frame.Image = copyOfImage;
615
616 // Add frame to frame list
617 lock (_cachedFrames)
618 {
619 _cachedFrames.Add(frame);
620 }
621 }

652 // Check if the next frame should be shown
653 private void UpdateFrameTime()
654 {
655 if (State != GifPlayerState.Playing) return; // Not playing
656
657 if (!Application.isPlaying || OverrideTimeScale)
658 {
659 // Play in editor mode or time is independant from Time.timeScale
660 if (OverrideTimeScale)
661 _secondsTillNextFrame -= (Time.realtimeSinceStartup - _editorPreviousUpdateTime) * TimeScale;
662 else
663 _secondsTillNextFrame -= Time.realtimeSinceStartup - _editorPreviousUpdateTime;
664 _editorPreviousUpdateTime = Time.realtimeSinceStartup;
665 return;
666 }
667
668 // Calculate seconds till next gif frame
669 _secondsTillNextFrame -= Time.deltaTime;
670 }
671
672 // Update the frame
673 private void UpdateFrame()
674 {
675 if (_gifDecoder.NumberOfFrames > 0 && _gifDecoder.NumberOfFrames == CurrentFrameNumber)
676 {
677 // Set frame number to 0 if we are at the last one
678 CurrentFrameNumber = 0;
679 if (!Loop)
680 {
681 // Stop playback if not looping
682 Pause();
683 return;
684 }
685 }
686
687 if (_cacheFrames)
688 {
689 // Frames are cached
690 lock (_cachedFrames)
691 {
692 CurrentFrame = _cachedFrames.Count > CurrentFrameNumber
693 ? _cachedFrames[CurrentFrameNumber]
694 : _gifDecoder.GetCurrentFrame();
695 }
696
697 // Prepare the next frame
698 if (!_gifDecoder.AllFramesRead)
699 // Not all frames are read yet. Prepare the next frame
700 StartReadFrame();
701 }
702 else
703 {
704 // Get the frame from the decoder
705 CurrentFrame = _gifDecoder.GetCurrentFrame();
706 }
707
708 // Update the target texture with the new frame
709 UpdateTexture();
710
711 // Set next frame time
712 SetNextFrameTime();
713
714 // Move to next frame
715 CurrentFrameNumber++;
716
717 if (!_cacheFrames) StartReadFrame(); // Prepare the next frame
718 }

721 // Check if the frame needs to be updated
722 private void CheckFrameChange()
723 {
724 if (State != GifPlayerState.Playing && _hasFirstFrameBeenShown || !_frameIsReady) return;
725
726 if (State == GifPlayerState.Loading) return;
727
728 if (!_hasFirstFrameBeenShown)
729 {
730 // Show the first frame
731 SetTexture();
732 lock (_locker)
733 {
734 UpdateFrame();
735 }
736
737 _hasFirstFrameBeenShown = true;
738 return;
739 }
740
741 UpdateFrameTime();
742
743 if (_secondsTillNextFrame > 0) return;
744
745 // Time to change the frame
746 lock (_locker)
747 {
748 UpdateFrame();
749 }
750 }

753 private void UpdateTexture()
754 {
755 // Upload texture data

757
758 // Apply
759 GifTexture.Apply();
760 }

762 // Starts reading a frame
763 private void StartReadFrame()
764 {
765 _frameIsReady = false;
766 if (_useThreadedDecoder)
767 {
768 // Signal thread to read the next frame
769 _wh.Set();
770 return;
771 }
772
773 // Not threaded
774 if (_cacheFrames && _gifDecoder.AllFramesRead) return; // Don't retrieve data if we already have cached all frames
775
776 // Read the next frame
777 _gifDecoder.ReadNextFrame(!_cacheFrames);
778
779 // Add frame to cache if caching is enabled
780 if (_cacheFrames && !_gifDecoder.AllFramesRead) AddFrameToCache(_gifDecoder.GetCurrentFrame());
781
782 // Mark the frame as ready
783 _frameIsReady = true;
784 }
785
786 // Start the decode thread
787 private void StartDecodeThread()
788 {
789#if !UNITY_WSA
790
791 if (!_useThreadedDecoder) return;
792
793
794 lock (_locker)
795 {
796 _threadLoop = !_cacheFrames;
797 _threadBufferAll = _bufferAllFrames;
798 }
799
800 if (_decodeThread != null) return; // Thread is already running
801
802 _threadIsCanceled = false;
803 _decodeThread = new Thread(FrameDataThread);
804 _decodeThread.Name = "gifDecoder" + _decodeThread.ManagedThreadId;
805 _decodeThread.IsBackground = true;
806 _decodeThread.Start();
807#endif
808 }
809
810 // Ends the decode thread. Used to clean up after application quit
811 private void EndDecodeThread()
812 {
813 if (_threadIsCanceled) return;
814 _threadIsCanceled = true;
815 _wh.Set();
816 }
817
818 // The decode thread
819 private void FrameDataThread()
820 {
821 _wh.WaitOne();
822 while (!_threadIsCanceled)
823 {
824 lock (_locker)
825 {
826 // Read the next frame
827 _gifDecoder.ReadNextFrame(_threadLoop);
828 if (_cacheFrames && _gifDecoder.AllFramesRead)
829 {
830 _frameIsReady = true;
831 break;
832 }
833
834 if (_cacheFrames) AddFrameToCache(_gifDecoder.GetCurrentFrame());
835
836 if (_threadBufferAll)
837 {
838 if (_gifDecoder.AllFramesRead)
839 {
840 _frameIsReady = true;
841 break;
842 }
843
844 continue;
845 }
846
847 _frameIsReady = true;
848 }
849
850 _wh.WaitOne(); // Wait for signal that frame must be read
851 }
852
853 _threadIsCanceled = true;
854 _decodeThread = null;
855 }
