74 var replant = false;

78 List<Block.SItemDropProb> harvestItems = new List<Block.SItemDropProb>();
79 var seedFromPlant = string.Empty;
80 if ( IsDeadPlant() )
81 {
82 plantData = new PlantData(plantPos);
83 }

86 // If we are a plant growing, then check to see if its the final stage.
87 if (currentBlock.Block.Properties.Contains("PlantGrowing.Next"))

90 return null;

94 currentBlock.Block.itemsToDrop.TryGetValue(EnumDropEvent.Harvest, out harvestItems);
95
96 // Including the breaking of the plant here.
97 currentBlock.Block.itemsToDrop.TryGetValue(EnumDropEvent.Destroy, out var harvestItems2);
98 if (harvestItems != null && harvestItems2 != null)
99 {
100 foreach (var item in harvestItems2)
101 harvestItems.Add(item);
102 }
103
104 // Clean all all the seeds
105 for (var x = harvestItems.Count - 1; x >= 0; x--)
106 {
107 var harvestItem = harvestItems[x];
108 if (harvestItem.name.StartsWith("planted"))
109 {
110 seedFromPlant = harvestItem.name;
111 harvestItems.RemoveAt(x);
112 }
113
114 // If the NPC has cvar of the name of the harvested item, then this will restrict the amount of items they get.
115 if ( entityAlive.Buffs.HasCustomVar(harvestItem.name) )
116 {
117 var count = entityAlive.Buffs.GetCustomVar(harvestItem.name);
118 harvestItem.minCount = (int)count;
119 harvestItem.maxCount = (int)count;
120 harvestItems[x] = harvestItem;
121 }
122 }
123
124 //GameManager.Instance.World.SetBlockRPC(plantPos, BlockValue.Air);
125 GameManager.Instance.World.SetBlock(0, plantPos, BlockValue.Air, false, false);
126 replant = true;
127 }
128 }
129 if (currentBlock.isair)
130 replant = true;
131
132 if ( replant )
133 {
134 // use the seed we got from the plant itself, if its set. If not, then search for one.
135 var seedName = seedFromPlant;
136 if (string.IsNullOrEmpty(seedName))

138 ItemStack itemStack = new ItemStack();
139 foreach (var stack in entityAlive.lootContainer.items)
140 {
141 if (stack.IsEmpty()) continue;
142 var itemname = stack.itemValue.ItemClass.GetItemName();
143 if (itemname.StartsWith("planted") && itemname.EndsWith("1"))
144 {
145 seedName = itemname;
146 itemStack = stack;
147 break;
148 }
149 }
150 }
151
152 // Still nothing to plant?
153 if (string.IsNullOrEmpty(seedName))
154 return harvestItems;
155
156 if (plantData == null)
157 plantData = new PlantData(BlockPos);
158
159 if (plantData != null)
160 {
161 if (!plantData.IsNearWater())
162 {
163 return harvestItems;

167 var blockValue = cropBlock.ToBlockValue();

175 blockValue.Block.PlaceBlock(GameManager.Instance.World, result, null);
