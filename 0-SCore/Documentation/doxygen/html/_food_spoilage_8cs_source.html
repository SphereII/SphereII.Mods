344 var itemClass = __instance.ItemStack.itemValue.ItemClass;
345 if (itemClass != null && itemClass.Properties.Contains(PropSpoilable) && itemClass.Properties.GetBool(PropSpoilable))

347 float DegradationMax = 1000f;
348 if (itemClass.Properties.Contains("SpoilageMax"))
349 DegradationMax = itemClass.Properties.GetFloat("SpoilageMax");
350
351 float PerCent = 1f - Mathf.Clamp01(__instance.ItemStack.itemValue.UseTimes / DegradationMax);
352 int TierColor = 7 + (int)Math.Round(8 * PerCent);
353 if (TierColor < 0)
354 TierColor = 0;
355 if (TierColor > 7)
356 TierColor = 7;
357
358 // allow over-riding of the color.
359 if (itemClass.Properties.Contains("QualityTierColor"))
360 TierColor = itemClass.Properties.GetInt("QualityTierColor");
361
362 // These used to be fields of the instance, not in A20
363 var controller = __instance.GetChildById("durability");
364 if (controller != null && controller.ViewComponent is XUiV_Sprite durability)
365 {
366 durability.IsVisible = true;
367 durability.Color = QualityInfo.GetQualityColor(TierColor);
368 durability.Fill = PerCent;
369 }
370
371 controller = __instance.GetChildById("durabilityBackground");
372 if (controller != null && controller.ViewComponent is XUiV_Sprite durabilityBackground)
373 {
374 durabilityBackground.IsVisible = true;
375 }
376 }
377 }
