1using Audio;
2using System.Collections.Generic;
3using UnityEngine;
4using UnityEngine.Video;
5using Random = System.Random;
6
7internal class MusicBoxScript : MonoBehaviour
8{
9 // This is the entity that initialized the music box.
10 public Entity myEntity;
11
12 // strSoundSource is the current SoundDataNode being played. It gets updated from the list of SoundGroups.
13 public string strSoundSource;
14
15 // strVideoSource is the current video clip being played. It gets updated from the list of video clips.
16 public string strVideoSource;
17
18 // These are the optional list of sound and videos clips that can be part of this.
19 public List<string> SoundGroups = new List<string>();
20 public List<string> VideoGroups = new List<string>();
21
22 // AudioSource is the SoundData's Node AudioSource attribute. It's the "speaker"
23 public string strAudioSource;
24
25
26 // References to our Animator and Video Player, if it exists.
27 public Animator anim;
28 public VideoPlayer videoPlayer;
29 public VideoClip defaultClip;
30
31 // The video clip reference, which gets loaded from the asset bundle.
32 public VideoClip myVideoClip;
33
34 // We need a randomize element to pick the sound and video from their groups.
35 public Random random = new Random();
36 public float CheckDelay = 1f;
37 public bool IsVideoPaused;
38 public Vector3i myBlockPos;
39
40 // Set up a throttle to delay the Update(), since we don't really need it called each frame
41 private float nextCheck;
42
43 private void Start()
44 {
45 if (videoPlayer && videoPlayer.clip != null)
46 defaultClip = videoPlayer.clip;
47
48 // myAudioSource = gameObject.AddComponent<AudioSource>();
49 }
50
51 // Update is called once per frame
52 private void Update()
53 {
54 if (nextCheck < Time.time)
55 {
56 nextCheck = Time.time + CheckDelay;
57 CheckAudioPlayer();
58 CheckVideoPlayer();
59 }
60 }
61
62 private void OnEnable()
63 {
64 // Toggle the animation if it's available
65 if (anim)
66 {
67 anim.enabled = true;
68 anim.SetBool("IsOn", true);
69 }
70 }
71
72 // Disable the script; turning off any sounds that may be playing, and shutting down the animation.
73 private void OnDisable()
74 {
75 if (videoPlayer)
76 videoPlayer.Stop();
77
78 if (anim)
79 anim.enabled = false;
80
81 // In order for BroadcastStop() to work, when you are using an external audio source, is that it must be set to Loop. This is the Unity AudioSource,
82 // not the sounds.xml AudioClip line.
83 Manager.BroadcastStop(myBlockPos.ToVector3(), strSoundSource);
84 }
85
86 private void CheckVideoPlayer()
87 {
88 if (!videoPlayer)
89 return;
90
91 if (GameManager.Instance.IsPaused())
92 {
93 videoPlayer.Pause();
94 IsVideoPaused = true;
95 return;
96 }
97
98 //videoPlayer.EnableAudioTrack(0, false);
99 if (IsVideoPaused)
100 {
101 // If the video was paused, and we want to resume it, just re-play, and not change any clips.
102 if (IsVideoPaused)
103 {
104 videoPlayer.Play();
105 IsVideoPaused = false;
106 return;
107 }
108 }
109
110 if (videoPlayer.isPlaying) return;
111
112 // If the video has a clip already attached, then check if there's another clip we could use instead.
113 if (videoPlayer.clip)
114 // Otherwise, if we have another video clip specified, use that.
115 if (myVideoClip)
116 videoPlayer.clip = myVideoClip;
117
118 // If there's still no video clip, try to parse the Video Source as a URL
119 if (!videoPlayer.clip) videoPlayer.url = strVideoSource;
120
121 // If the VideoGroups is populated, that means we can randomize based on all the videos groups in our container.
122 if (VideoGroups.Count > 0)
123 {
124 var randomIndex = random.Next(0, VideoGroups.Count);
125 strVideoSource = VideoGroups[randomIndex];
126 if (strVideoSource.StartsWith("#")) videoPlayer.clip = DataLoader.LoadAsset<VideoClip>(strVideoSource);
127 }
128 else
129 {
130 if (defaultClip)
131 videoPlayer.clip = defaultClip;
132 else
133 Debug.Log("There are no Groups of video detected.");
134 }
135
136 videoPlayer.Play();
137 }
138
139 private void CheckAudioPlayer()
140 {
141 // Check to see if we are still playing music or not, and if everything is initialized.
142 if (!IsPlaying())
143 // Toggle the sound if we have sound references.
144 if (myEntity && !string.IsNullOrEmpty(strSoundSource) && !string.IsNullOrEmpty(strAudioSource))
145 {
146 // If the SoundGroups is populated, that means we can randomize based on all the sound groups in our container.
147 if (SoundGroups.Count > 0)
148 {
149 var randomIndex = random.Next(0, SoundGroups.Count);
150 strSoundSource = SoundGroups[randomIndex];
151 }
152
153 Manager.BroadcastPlay(myBlockPos.ToVector3(), strSoundSource);
154 }
155 }
156
157 // Since the Manager.Audio doesn't give us access to our playing source, we have to poll the List<> of sources to see if ours is still playing.
158 private bool IsPlaying()
159 {
160 // If the audio source isn't set, don't blindly try to play, but tell the loop that something is playing to prevent it starting something.
161 if (string.IsNullOrEmpty(strAudioSource))
162 return true;
163
164 // look for the audio sources
165 foreach (var each in Manager.playingAudioSources)
166 if (each.ToString().Contains(strAudioSource))
167 return true;
168
169 return false;
170 }
171}
