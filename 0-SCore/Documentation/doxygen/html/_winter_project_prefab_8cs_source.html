162 public static void SetSnow(int startX, int startZ, int width, int depth, ChunkCluster cluster, bool notifyRpc,
163 bool log, bool AlreadyFilled, int height)
164 {
165 Write("Set snow " + startX + "," + startZ + " by " + width + "," + depth);
166
167 //WorldBase world = cluster.GetWorld();
168 var chunk = cluster.GetChunkSync(World.toChunkXZ(startX), World.toChunkXZ(startZ));
169
170 var worldPos = new Vector3i(startX, 0, startZ);
171 var size = new Vector3i(width, height, depth);
172
173 var processed = new List<long>();
174
175 for (var x = 0; x < width; x++)
176 {
177 for (var z = 0; z <= depth; z++)
178 {
179 var worldX = x + startX;
180 var worldZ = z + startZ;
181 var chunkWorldX = World.toChunkXZ(worldX);
182 var chunkWorldZ = World.toChunkXZ(worldZ);
183 //var chunkX = World.toBlockXZ(worldX);
184 //var chunkZ = World.toBlockXZ(worldZ);
185 if (chunk == null || chunk.X != chunkWorldX || chunk.Z != chunkWorldZ)
186 {
187 Write("Getting chunk at " + chunkWorldX + "," + chunkWorldZ + " " + worldX + "," + worldZ);
188 chunk = cluster.GetChunkSync(chunkWorldX, chunkWorldZ);
189
190 if (chunk != null)
191 {
192 if (processed.Contains(chunk.Key))
193 continue;
194 processed.Add(chunk.Key);
195 ProcessChunk(chunk, worldPos, size, AlreadyFilled, log, notifyRpc, true);
196 }
197 }
198 }
199 }

201 Write("Set Snow Complete");
202 }
