47 public override void Update(Context _context)

49 // We don't call base.Update() here, because it's checking if the entity has no path, then it stops the task.

66 private void CheckProximityToPosition(Context context)
67 {
68 context.Self.SetLookPosition(Vector3.zero);
69 if (context.ActionData.Target is Vector3)
70 {
71 if (context.Self.getNavigator().noPathAndNotPlanningOne())
72 Stop(context);
73
74 var distance = Vector3.Distance(context.Self.position, _position);
75 if (distance < 2f)
76 {
77 Stop(context);
78 return;
79 }
80 }
81
82 var entityAlive = UAIUtils.ConvertToEntityAlive(context.ActionData.Target);
83 if (entityAlive == null) return;
84 if (entityAlive.IsDead())
85 {
86 Stop(context);
87 return;
88 }
89
90 // The entity hasn't moved very much, keep going.
91 if (Vector3.Distance(entityAlive.position, _position) < 2f)
92 {
93 context.Self.SetLookPosition(_position);
94 context.Self.RotateTo(entityAlive, 30f, 30f);
95 context.Self.moveHelper.SetMoveTo(_position, true);
96 return;
97 }
98
99 // We are close enough
100 if (Vector3.Distance(context.Self.position, entityAlive.position) < 1.2)
101 {
102 Stop(context);
103 return;
104 }
105
106 _position = entityAlive.position;
107 SCoreUtils.FindPath(context, _position, run);
