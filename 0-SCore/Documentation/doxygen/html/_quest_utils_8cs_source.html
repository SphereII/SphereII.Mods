39 public static PrefabInstance FindClosesPrefabs(Vector3 position, List<PrefabInstance> prefabs, List<Vector2> usedPOILocations, BiomeFilterTypes biomeFilterType, string biomeFilter)
40 {
41 PrefabInstance prefab = null;
42 float minDist = Mathf.Infinity;
43 string[] array = null;
44
45 foreach (var t in prefabs)
46 {
47 // Have we already went to this one?
48 Vector2 vector = new Vector2((float)t.boundingBoxPosition.x, (float)t.boundingBoxPosition.z);
49 if (usedPOILocations != null && usedPOILocations.Contains(vector))
50 continue;
51
52 // Check if there's a biome filter.
53 if (biomeFilterType != BiomeFilterTypes.AnyBiome)
54 {
55 BiomeDefinition biomeAt = GameManager.Instance.World.ChunkCache.ChunkProvider.GetBiomeProvider().GetBiomeAt((int)vector.x, (int)vector.y);
56 if (biomeFilterType == BiomeFilterTypes.OnlyBiome && biomeAt.m_sBiomeName != biomeFilter)
57 {
58 if (GamePrefs.GetBool(EnumGamePrefs.DebugMenuEnabled))
59 Log.Out($"GotoPOISDX: Prefab Filtered based on Biome Filter Type: {biomeFilterType} Biome Filter: {biomeFilter}");
60
61 continue;
62 }
63 if (biomeFilterType == BiomeFilterTypes.ExcludeBiome)
64 {
65 if (array == null)
66 {
67 array = biomeFilter.Split(new char[]
68 {
69 ','
70 });
71 }
72 bool flag = false;
73 for (int j = 0; j < array.Length; j++)
74 {
75 if (biomeAt.m_sBiomeName == array[j])
76 {
77 if (GamePrefs.GetBool(EnumGamePrefs.DebugMenuEnabled))
78 Log.Out($"GotoPOISDX: Prefab excluded based on Biome Filter Type: {biomeFilterType} Biome Filter: {biomeFilter}");
79
80 flag = true;
81 break;
82 }
83 }
84 if (flag)
85 {
86 continue;
87 }
88 }
89 }
90
91 float dist = Vector3.Distance(t.boundingBoxPosition, position);
92 if (GamePrefs.GetBool(EnumGamePrefs.DebugMenuEnabled))
93 Log.Out($"GotoPOISDX: Prefab {t.name} Found at {t.boundingBoxPosition} Distance: {dist} {biomeFilterType} Biome Filter: {biomeFilter}");
94
95 if (dist < minDist)
96 {
97 if (GamePrefs.GetBool(EnumGamePrefs.DebugMenuEnabled))
98 if (prefab != null)
99 Log.Out($"GotoPOISDX: Found closer Prefab {t.name} than {prefab.name} Old distance {minDist}");
100 prefab = t;
101 minDist = dist;
102 }
103 }
104
105 return prefab;
106 }

134 public static PrefabInstance GetRandomPOINearTrader(
135 EntityTrader trader,
136 FastTags questTag,
137 byte difficulty,
138 POITags includeTags,
139 POITags excludeTags,
140 float minSearchDistance,
141 float maxSearchDistance,
142 List<Vector2> usedPoiLocations = null,
143 int entityIdForQuests = -1,
144 BiomeFilterTypes biomeFilterType = BiomeFilterTypes.AnyBiome,
145 string biomeFilter = "")
146 {
147 World world = GameManager.Instance.World;
148
149 int minDistanceTier = minSearchDistance < 0 ? 0 : GetTraderPrefabListTier(minSearchDistance);
150 int maxDistanceTier = maxSearchDistance < 0 ? 2 : GetTraderPrefabListTier(maxSearchDistance);
151
152 for (int distanceTier = minDistanceTier; distanceTier <= maxDistanceTier; distanceTier++)
153 {
154 List<PrefabInstance> prefabsForTrader = QuestEventManager.Current.GetPrefabsForTrader(
155 trader.traderArea,
156 difficulty,
157 distanceTier,
158 world.GetGameRandom());
159
160 if (prefabsForTrader != null)
161 {
162 // GetPrefabsForTrader shuffles the prefabs before returning them, so we can just
163 // iterate through the list and still send players to "random" POIs
164 for (int j = 0; j < prefabsForTrader.Count; j++)
165 {
166 PrefabInstance prefabInstance = prefabsForTrader[j];
167 if (ValidPrefabForQuest(
168 trader,
169 prefabInstance,
170 questTag,
171 includeTags,
172 excludeTags,
173 usedPoiLocations,
174 entityIdForQuests,
175 biomeFilterType,
176 biomeFilter,
177 minSearchDistance,
178 maxSearchDistance))
179 {
180 return prefabInstance;
181 }
182 }
183 }
184 }
185
186 return null;

288 public static bool ValidPrefabForQuest(
289 Entity questGiver,
290 PrefabInstance prefab,
291 FastTags questTag,
292 POITags includeTags,
293 POITags excludeTags,
294 List<Vector2> usedPoiLocations = null,
295 int entityIdForQuests = -1,
296 BiomeFilterTypes biomeFilterType = BiomeFilterTypes.AnyBiome,
297 string biomeFilter = "",
298 float minSearchDistance = -1,
299 float maxSearchDistance = -1)
300 {
301 if (!prefab.prefab.bSleeperVolumes)
302 {
303 if (LoggingEnabled)
304 {
305 Log.Out($"Quest {questTag}: Prefab {prefab.name} has no sleeper volumes");
306 }
307 return false;
308 }
309
310 //if (!prefab.prefab.GetQuestTag(questTag))
311 //{
312 // if (LoggingEnabled)
313 // {
314 // Log.Out($"Quest {questTag}: Prefab {prefab.name} does not have quest tag {questTag}");
315 // }
316 // return false;
317 //}
318
319 Vector2 poiLocation = new Vector2(prefab.boundingBoxPosition.x, prefab.boundingBoxPosition.z);
320
321 if (usedPoiLocations != null && usedPoiLocations.Contains(poiLocation))
322 {
323 if (LoggingEnabled)
324 {
325 Log.Out($"Quest {questTag}: Prefab {prefab.name} has already been used");
326 }
327 return false;
328 }
329
330 QuestEventManager.POILockoutReasonTypes lockoutReason = QuestEventManager.Current.CheckForPOILockouts(entityIdForQuests, poiLocation, out var num);
331
332 if (lockoutReason != QuestEventManager.POILockoutReasonTypes.None)
333 {
334 if (LoggingEnabled)
335 {
336 Log.Out($"Quest {questTag}: Prefab {prefab.name} is locked out: {lockoutReason}");
337 }
338 return false;
339 }
340
341 if (!MeetsBiomeRequirements(poiLocation, questGiver, biomeFilterType, biomeFilter))
342 {
343 if (LoggingEnabled)
344 {
345 Log.Out($"Quest {questTag}: Prefab {prefab.name} does not meet biome requirements: type={biomeFilterType}, filter={biomeFilter}");
346 }
347 return false;
348 }
349
350 if (!MeetsTagRequirements(prefab, includeTags, excludeTags))
351 {
352 if (LoggingEnabled)
353 {
354 var includeTagsText = includeTags.IsEmpty ? string.Empty : $"must include {includeTags}; ";
355 var excludeTagsText = excludeTags.IsEmpty ? string.Empty : $"must exclude {excludeTags}; ";
356 Log.Out($"Quest {questTag}: Prefab {prefab.name} fails tag requirements: {includeTagsText}{excludeTagsText}prefab tags are{prefab.prefab.Tags}");
357 }
358 return false;
359 }
360
361 if (!MeetsDistanceRequirements(prefab, questGiver, minSearchDistance, maxSearchDistance))
362 {
363 if (LoggingEnabled)
364 {
365 var minSearchText = minSearchDistance > 0 ? $" minimum distance {minSearchDistance} and" : string.Empty;
366 Log.Out($"Quest {questTag}: Prefab {prefab.name} is not within{minSearchText} maximum distance {maxSearchDistance}");
367 }
368 return false;
369 }
370
371 return true;
372 }

383 public static bool MeetsDistanceRequirements(
384 PrefabInstance prefab,
385 Entity questGiver,
386 float minSearchDistance,
387 float maxSearchDistance)
388 {
389 if (minSearchDistance < 0 && maxSearchDistance < 0)
390 return true;
391
392 Vector2 worldPos = new Vector2(questGiver.position.x, questGiver.position.z);
393
394 Vector2 prefabCenter = prefab.GetCenterXZ();
395
396 // Work with the square of the distance to avoid taking square roots
397 float sqrDistance = (worldPos - prefabCenter).sqrMagnitude;
398
399 if (minSearchDistance >= 0 &&
400 sqrDistance < (minSearchDistance * minSearchDistance))
401 {
402 return false;
403 }
404
405 if (maxSearchDistance >= 0 &&
406 sqrDistance > (maxSearchDistance * maxSearchDistance))
407 {
408 return false;
409 }
410
411 return true;
412 }

422 public static bool MeetsBiomeRequirements(
423 Vector2 poiLocation,
424 Entity questGiver,
425 BiomeFilterTypes biomeFilterType,
426 string biomeFilter)
427 {
428 if (biomeFilterType == BiomeFilterTypes.AnyBiome)
429 {
430 return true;
431 }
432
433 IBiomeProvider biomeProvider = GameManager.Instance.World.ChunkCache.ChunkProvider.GetBiomeProvider();
434
435 BiomeDefinition poiBiome = biomeProvider.GetBiomeAt((int)poiLocation.x, (int)poiLocation.y);
436
437 if (biomeFilterType == BiomeFilterTypes.OnlyBiome && poiBiome.m_sBiomeName != biomeFilter)
438 {
439 return false;
440 }
441 else if (biomeFilterType == BiomeFilterTypes.ExcludeBiome)
442 {
443 string[] biomes = biomeFilter.Split(new char[] { ',' });
444
445 bool inBiome = false;
446
447 for (int i = 0; i < biomes.Length; i++)
448 {
449 if (poiBiome.m_sBiomeName == biomes[i])
450 {
451 inBiome = true;
452 break;
453 }
454 }
455
456 if (inBiome)
457 {
458 return false;
459 }
460 }
461 else if (biomeFilterType == BiomeFilterTypes.SameBiome && questGiver != null)
462 {
463 var questGiverBiome = biomeProvider.GetBiomeAt(
464 (int)questGiver.position.x,
465 (int)questGiver.position.z);
466
467 if (poiBiome != questGiverBiome && biomeFilter != "" && poiBiome.m_sBiomeName != biomeFilter)
468 {
469 return false;
470 }
471 }
472
473 return true;
474 }

485 public static bool MeetsTagRequirements(PrefabInstance prefabInstance, POITags includeTags, POITags excludeTags)
486 {
487 if (!includeTags.IsEmpty &&

489 {

491 }
492
493 if (!excludeTags.IsEmpty &&
494 prefabInstance.prefab.Tags.Test_AnySet(excludeTags))

496 return false;

499 return true;
